   27 -- | All runtime dependencies for the agent loop, gathered into a single record.
   28 -- This replaces the multi-parameter signature of 'runAgentLoop' and
   29 -- 'executeSlashCommand', making it easy to add new capabilities (e.g.
   30 -- 'VaultHandle') in later work units without touching call sites.
   31 data AgentEnv = AgentEnv
   32   { _env_provider     :: IORef (Maybe SomeProvider)
   33     -- ^ The LLM provider. 'Nothing' when no credentials are configured yet.
   34   , _env_model        :: IORef ModelId
   35     -- ^ The model to use for completions. Mutable so slash commands
   36     -- like @\/provider@ and @\/target@ can hot-swap it.
   37   , _env_channel      :: ChannelHandle
   38     -- ^ The channel to read messages from and write responses to.
   39   , _env_logger       :: LogHandle
   40     -- ^ Structured logger for diagnostic output.
   41   , _env_systemPrompt :: Maybe Text
   42     -- ^ Optional system prompt prepended to every conversation.
   43   , _env_registry     :: ToolRegistry
   44     -- ^ All registered tools available for the agent to call.
   45   , _env_vault        :: IORef (Maybe VaultHandle)
   46     -- ^ Optional secrets vault. 'Nothing' if no vault is configured.
   47   , _env_pluginHandle :: PluginHandle
   48     -- ^ Handle for detecting and generating age plugin identities.
   49   , _env_policy :: SecurityPolicy
   50     -- ^ Security policy for command authorization. Needed by harness management.
   51   , _env_harnesses :: IORef (Map Text HarnessHandle)
   52     -- ^ Running harness handles, keyed by name (e.g. "claude-code").
   53   , _env_target :: IORef MessageTarget
   54     -- ^ Where incoming user messages are routed. Mutable so @\/target@
   55     -- can hot-swap the destination.
   56   , _env_nextWindowIdx :: IORef Int
   57     -- ^ Monotonically increasing counter for assigning tmux window indices
   58     -- to new harnesses. Starts at 0.
   59   , _env_agentDef :: Maybe AgentDef
   60     -- ^ Currently-selected agent, if any. Populated by the @--agent@ flag
   61     -- or the @default_agent@ config field. Used by agent-aware slash
   62     -- commands; 'Nothing' in the backward-compat no-agent path.
   63   , _env_session :: IORef SessionHandle
   64     -- ^ Current conversation session. Mutable so @\/session new@ and
   65     -- @\/session resume@ can swap the active session in place.
   66   , _env_onFirstStreamDone :: IORef (Maybe (IO ()))
   67     -- ^ One-shot callback fired by 'runAgentLoop' after the first
   68     -- 'StreamDone' it observes. The loop atomically consumes the
   69     -- action (setting the field back to 'Nothing') so a second
   70     -- 'StreamDone' does not re-fire it. In production this is
   71     -- populated with @'markBootstrapConsumed' session@ so the agent
   72     -- marks its bootstrap as consumed exactly once per process start.
   73   }
   74 
   75 -- | Read the active session's transcript handle.
   76 --
   77 -- The transcript lives inside the session directory (@transcript.jsonl@)
   78 -- and is swapped automatically when @\/session new@ or
   79 -- @\/session resume@ replaces the active session. Callers that previously
   80 -- read @_env_transcript@ should use this accessor instead.
   81 envTranscript :: AgentEnv -> IO TranscriptHandle
   82 envTranscript env = _sh_transcript <$> readIORef (_env_session env)
