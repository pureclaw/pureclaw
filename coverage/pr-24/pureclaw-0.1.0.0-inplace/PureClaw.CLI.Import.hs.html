   35 -- ---------------------------------------------------------------------------
   36 -- JSON5 preprocessor
   37 -- ---------------------------------------------------------------------------
   38 
   39 -- | Strip JSON5 features (// comments, trailing commas) to produce valid JSON.
   40 -- Handles comments inside strings correctly (does not strip them).
   41 -- Does NOT handle: block comments, hex literals, multiline strings,
   42 -- unquoted keys, or other advanced JSON5 features.
   43 stripJson5 :: Text -> Text
   44 stripJson5 = T.pack . go False . T.unpack
   45   where
   46     go :: Bool -> String -> String
   47     go _ [] = []
   48     go True ('\\' : c : rest) = '\\' : c : go True rest
   49     go True ('"' : rest) = '"' : go False rest
   50     go True (c : rest) = c : go True rest
   51     go False ('"' : rest) = '"' : go True rest
   52     go False ('/' : '/' : rest) = go False (dropWhile (/= '\n') rest)
   53     go False (',' : rest)
   54       | trailingComma rest = go False rest
   55     go False (c : rest) = c : go False rest
   56 
   57     trailingComma :: String -> Bool
   58     trailingComma [] = True
   59     trailingComma (c : rest)
   60       | c `elem` (" \t\n\r" :: String) = trailingComma rest
   61       | c == ']' || c == '}' = True
   62       | otherwise = False

  104 parseOpenClawConfig :: Value -> Either String OpenClawConfig
  105 parseOpenClawConfig = parseEither parseOC
  106 
  107 parseOC :: Value -> Parser OpenClawConfig
  108 parseOC = withObject "OpenClawConfig" $ \o -> do
  109   mAgents <- o .:? "agents"
  110   defaults <- maybe (pure emptyDefaults) parseDefaults mAgents
  111   agents <- maybe (pure []) parseAgentList mAgents
  112   mChannels <- o .:? "channels"
  113   signal <- maybe (pure Nothing) (withObject "channels" (.:? "signal") >=> traverse parseSignalCfg) mChannels
  114   telegram <- maybe (pure Nothing) (withObject "channels" (.:? "telegram") >=> traverse parseTelegramCfg) mChannels
  115   pure OpenClawConfig
  116     { _oc_defaultModel = fst defaults
  117     , _oc_workspace    = snd defaults
  118     , _oc_agents       = agents
  119     , _oc_signal       = signal
  120     , _oc_telegram     = telegram
  121     }
  122 
  123 emptyDefaults :: (Maybe Text, Maybe Text)
  124 emptyDefaults = (Nothing, Nothing)
  125 
  126 parseDefaults :: Value -> Parser (Maybe Text, Maybe Text)
  127 parseDefaults = withObject "agents" $ \o -> do
  128   mDefaults <- o .:? "defaults"
  129   case mDefaults of
  130     Nothing -> pure emptyDefaults
  131     Just defVal -> flip (withObject "defaults") defVal $ \d -> do
  132       mModelVal <- d .:? "model"
  133       model <- case mModelVal of
  134         Just (Object m) -> m .:? "primary"
  135         Just (String s) -> pure (Just s)
  136         _               -> pure Nothing
  137       ws <- d .:? "workspace"
  138       pure (model, ws)
  139 
  140 parseAgentList :: Value -> Parser [OpenClawAgent]
  141 parseAgentList = withObject "agents" $ \o -> do
  142   mList <- o .:? "list"
  143   case mList of
  144     Nothing -> pure []
  145     Just agents -> mapM parseAgentDef agents
  146 
  147 parseAgentDef :: Value -> Parser OpenClawAgent
  148 parseAgentDef = withObject "OpenClawAgent" $ \o -> do
  149   agentId <- o .: "id"
  150   systemPrompt <- o .:? "systemPrompt"
  151   mModelVal <- o .:? "model"
  152   let model = case mModelVal of
  153         Just (Object m) -> parseMaybe (.: "primary") m
  154         Just (String s) -> Just s
  155         _               -> Nothing
  156   mTools <- o .:? "tools"
  157   let toolProfile = mTools >>= parseMaybe (withObject "tools" (.: "profile"))
  158   ws <- o .:? "workspace"
  159   pure OpenClawAgent
  160     { _oca_id           = agentId
  161     , _oca_systemPrompt = systemPrompt
  162     , _oca_model        = model
  163     , _oca_toolProfile  = toolProfile
  164     , _oca_workspace    = ws
  165     }
  166 
  167 parseSignalCfg :: Value -> Parser OpenClawSignal
  168 parseSignalCfg = withObject "signal" $ \o ->
  169   OpenClawSignal <$> o .:? "account" <*> o .:? "dmPolicy" <*> o .:? "allowFrom"

  172 parseTelegramCfg = withObject "telegram" $ \o ->

  179 -- | Resolve $include directives in a JSON Value, up to a max depth.
  180 resolveIncludes :: Int -> FilePath -> Value -> IO Value
  181 resolveIncludes maxDepth baseDir val
  182   | maxDepth <= 0 = pure val
  183   | otherwise = case val of
  184       Object o -> case KM.lookup (Key.fromText "$include") o of
  185         Just (String path) -> do
  186           let fullPath = baseDir </> T.unpack path
  187           included <- loadJson5File fullPath
  188           case included of
  189             Left _    -> pure val
  190             Right inc -> resolveIncludes (maxDepth - 1) (takeDirectory fullPath) inc
  191         Just (Array paths) -> do
  192           resolved <- mapM (resolveIncludePath (maxDepth - 1) baseDir) (V.toList paths)
  193           pure (foldl deepMerge (Object KM.empty) resolved)
  194         _ -> do
  195           resolved <- KM.traverseWithKey (\_ v -> resolveIncludes maxDepth baseDir v) o
  196           pure (Object resolved)
  197       _ -> pure val
  198 
  199 resolveIncludePath :: Int -> FilePath -> Value -> IO Value
  200 resolveIncludePath depth baseDir (String path) = do
  201   let fullPath = baseDir </> T.unpack path
  202   loaded <- loadJson5File fullPath
  203   case loaded of
  204     Left _    -> pure (Object KM.empty)
  205     Right inc -> resolveIncludes depth (takeDirectory fullPath) inc
  206 resolveIncludePath _ _ other = pure other
  207 
  208 deepMerge :: Value -> Value -> Value
  209 deepMerge (Object a) (Object b) = Object (KM.unionWith deepMerge a b)
  210 deepMerge _ b = b
  211 
  212 -- | Load and parse a JSON5 file.
  213 loadJson5File :: FilePath -> IO (Either String Value)
  214 loadJson5File path = do
  215   result <- try @IOException (TIO.readFile path)
  216   case result of
  217     Left err -> pure (Left (show err))
  218     Right text ->
  219       let cleaned = stripJson5 text
  220           bs = TE.encodeUtf8 cleaned
  221       in pure (eitherDecodeStrict' bs)

  235 -- | Import an OpenClaw config file into PureClaw's directory structure.
  236 -- Writes to @configDir/config.toml@ and @configDir/agents/*/AGENTS.md@.
  237 importOpenClawConfig :: FilePath -> FilePath -> IO (Either Text ImportResult)
  238 importOpenClawConfig openclawPath configDir = do
  239   loaded <- loadJson5File openclawPath
  240   case loaded of
  241     Left err -> pure (Left ("Failed to parse OpenClaw config: " <> T.pack err))
  242     Right rawJson -> do
  243       resolved <- resolveIncludes 3 (takeDirectory openclawPath) rawJson
  244       case parseOpenClawConfig resolved of
  245         Left err -> pure (Left ("Failed to extract config fields: " <> T.pack err))
  246         Right ocConfig -> writeImportedConfig configDir ocConfig
  247 
  248 writeImportedConfig :: FilePath -> OpenClawConfig -> IO (Either Text ImportResult)
  249 writeImportedConfig configDir ocConfig = do
  250   Dir.createDirectoryIfMissing True configDir
  251   let agentsDir = configDir </> "agents"
  252 
  253   -- Write main config.toml
  254   let configContent = buildConfigToml ocConfig
  255   TIO.writeFile (configDir </> "config.toml") configContent
  256 
  257   -- Write agent AGENTS.md files
  258   agentNames <- mapM (writeAgentFile agentsDir) (_oc_agents ocConfig)
  259 
  260   -- Write default agent if there are defaults
  261   case (_oc_defaultModel ocConfig, _oc_workspace ocConfig) of
  262     (Nothing, Nothing) -> pure ()
  263     _ -> do
  264       let defaultDir = agentsDir </> "default"
  265       Dir.createDirectoryIfMissing True defaultDir
  266       TIO.writeFile (defaultDir </> "AGENTS.md") $ T.unlines
  267         [ "---"
  268         , maybe "" ("model: " <>) (_oc_defaultModel ocConfig)
  269         , maybe "" ("workspace: " <>) (_oc_workspace ocConfig)
  270         , "---"
  271         , ""
  272         , "Default PureClaw agent."
  273         ]
  274 
  275   pure (Right ImportResult
  276     { _ir_configWritten = True
  277     , _ir_agentsWritten = agentNames
  278     , _ir_skippedFields = []
  279     , _ir_warnings      = []
  280     })

  282 buildConfigToml :: OpenClawConfig -> Text
  283 buildConfigToml oc = T.unlines $ concatMap (filter (not . T.null))
  284   [ maybe [] (\m -> ["model = " <> quoted m]) (_oc_defaultModel oc)
  285   , case _oc_signal oc of
  286       Nothing -> []
  287       Just sig ->
  288         [ ""
  289         , "[signal]"
  290         ] ++ catMaybes
  291         [ fmap (\a -> "account = " <> quoted a) (_ocs_account sig)
  292         , fmap (\p -> "dm_policy = " <> quoted (camelToSnake p)) (_ocs_dmPolicy sig)
  293         , fmap (\af -> "allow_from = " <> fmtList af) (_ocs_allowFrom sig)
  294         ]
  295   , case _oc_telegram oc of
  296       Nothing -> []
  297       Just tg ->
  298         [ ""
  299         , "[telegram]"
  300         ] ++ catMaybes
  301         [ fmap (\t -> "bot_token = " <> quoted t) (_oct_botToken tg)
  302         , fmap (\p -> "dm_policy = " <> quoted (camelToSnake p)) (_oct_dmPolicy tg)
  303         , fmap (\af -> "allow_from = " <> fmtList af) (_oct_allowFrom tg)
  304         ]
  305   ]
  306   where
  307     catMaybes = foldr (\x acc -> maybe acc (: acc) x) []

  309 writeAgentFile :: FilePath -> OpenClawAgent -> IO Text
  310 writeAgentFile agentsDir agent = do
  311   let agentDir = agentsDir </> T.unpack (_oca_id agent)
  312   Dir.createDirectoryIfMissing True agentDir
  313   let frontmatterLines = filter (/= "")
  314         [ maybe "" ("model: " <>) (_oca_model agent)
  315         , maybe "" ("tool_profile: " <>) (_oca_toolProfile agent)
  316         , maybe "" ("workspace: " <>) (_oca_workspace agent)
  317         ]
  318       hasFrontmatter = not (null frontmatterLines)
  319       header = if hasFrontmatter
  320         then ["---"] ++ frontmatterLines ++ ["---", ""]
  321         else []
  322       body = fromMaybe "" (_oca_systemPrompt agent)
  323   TIO.writeFile (agentDir </> "AGENTS.md") (T.unlines (header ++ [body]))
  324   pure (_oca_id agent)
  325 
  326 quoted :: Text -> Text
  327 quoted t = "\"" <> T.replace "\"" "\\\"" t <> "\""
  328 
  329 fmtList :: [Text] -> Text
  330 fmtList xs = "[" <> T.intercalate ", " (map quoted xs) <> "]"
  331 
  332 camelToSnake :: Text -> Text
  333 camelToSnake = T.concatMap $ \c ->
  334   if Char.isAsciiUpper c
  335     then T.pack ['_', Char.toLower c]
  336     else T.singleton c
