   50 ollamaComplete :: OllamaProvider -> CompletionRequest -> IO CompletionResponse
   51 ollamaComplete provider req = do
   52   initReq <- HTTP.parseRequest (_ol_baseUrl provider)
   53   let httpReq = initReq
   54         { HTTP.method = "POST"
   55         , HTTP.requestBody = HTTP.RequestBodyLBS (encodeRequest req)
   56         , HTTP.requestHeaders = [("content-type", "application/json")]
   57         }
   58   resp <- HTTP.httpLbs httpReq (_ol_manager provider)
   59   let status = Status.statusCode (HTTP.responseStatus resp)
   60   if status /= 200
   61     then throwIO (OllamaAPIError status (BL.toStrict (HTTP.responseBody resp)))
   62     else case decodeResponse (HTTP.responseBody resp) of
   63       Left err -> throwIO (OllamaParseError (T.pack err))
   64       Right response -> pure response

   66 -- | Encode a request for the Ollama /api/chat endpoint.
   67 -- Ollama uses system messages in the messages array and a simpler
   68 -- tool format than OpenAI.
   69 encodeRequest :: CompletionRequest -> BL.ByteString
   70 encodeRequest req = encode $ object $
   71   [ "model"    .= unModelId (_cr_model req)
   72   , "messages" .= encodeMessages req
   73   , "stream"   .= False
   74   ]
   75   ++ ["tools" .= map encodeTool (_cr_tools req) | not (null (_cr_tools req))]
   76 
   77 encodeMessages :: CompletionRequest -> [Value]
   78 encodeMessages req =
   79   maybe [] (\s -> [object ["role" .= ("system" :: Text), "content" .= s]]) (_cr_systemPrompt req)
   80   ++ map encodeMsg (_cr_messages req)
   81 
   82 encodeMsg :: Message -> Value
   83 encodeMsg msg = case _msg_content msg of
   84   [TextBlock t] ->
   85     object ["role" .= roleToText (_msg_role msg), "content" .= t]
   86   blocks ->
   87     -- Ollama supports content as string only; concatenate text blocks
   88     let textParts = concatMap blockText blocks
   89     in object ["role" .= roleToText (_msg_role msg), "content" .= T.intercalate "\n" textParts]
   90 
   91 blockText :: ContentBlock -> [Text]
   92 blockText (TextBlock t) = [t]
   93 blockText (ImageBlock _ _) = ["[image]"]
   94 blockText (ToolUseBlock _ name _) = ["[tool:" <> name <> "]"]
   95 blockText (ToolResultBlock _ parts _) = [t | TRPText t <- parts]
   96 
   97 encodeTool :: ToolDefinition -> Value
   98 encodeTool td = object
   99   [ "type" .= ("function" :: Text)
  100   , "function" .= object
  101       [ "name"        .= _td_name td
  102       , "description" .= _td_description td
  103       , "parameters"  .= _td_inputSchema td
  104       ]
  105   ]

  107 -- | Decode an Ollama /api/chat response.
  108 decodeResponse :: BL.ByteString -> Either String CompletionResponse
  109 decodeResponse bs = eitherDecode bs >>= parseEither parseResp
  110   where
  111     parseResp :: Value -> Parser CompletionResponse
  112     parseResp = withObject "OllamaResponse" $ \o -> do
  113       msg <- o .: "message"
  114       content <- msg .: "content"
  115       modelText <- o .: "model"
  116       -- Ollama tool calls come as tool_calls array in the message
  117       toolCalls <- msg .:? "tool_calls" .!= ([] :: [Value])
  118       toolBlocks <- mapM parseToolCall toolCalls
  119       let textBlocks = [TextBlock content | not (T.null content)]
  120       pure CompletionResponse
  121         { _crsp_content = textBlocks ++ toolBlocks
  122         , _crsp_model   = ModelId modelText
  123         , _crsp_usage   = Nothing  -- Ollama doesn't report usage in chat endpoint
  124         }
  125 
  126     parseToolCall :: Value -> Parser ContentBlock
  127     parseToolCall = withObject "ToolCall" $ \tc -> do
  128       fn <- tc .: "function"
  129       name <- fn .: "name"
  130       args <- fn .: "arguments"
  131       -- Ollama doesn't return a call ID, so generate a placeholder
  132       pure (ToolUseBlock (ToolCallId ("ollama-" <> name)) name args)
