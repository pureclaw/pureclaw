   30 -- | Run the main agent loop. Reads messages from the channel, sends
   31 -- them to the provider (with tool definitions), handles tool call/result
   32 -- cycles, and writes responses back.
   33 --
   34 -- Slash commands (messages starting with '/') are intercepted and
   35 -- handled before being sent to the provider.
   36 --
   37 -- If no provider is configured ('Nothing' in the IORef), chat messages
   38 -- produce a helpful error directing the user to configure credentials.
   39 -- Slash commands always work regardless of provider state.
   40 --
   41 -- Exits cleanly on 'IOException' from the channel (e.g. EOF / Ctrl-D).
   42 -- Provider errors are logged and a 'PublicError' is sent to the channel.
   43 runAgentLoop :: AgentEnv -> IO ()
   44 runAgentLoop env = do
   45   _lh_logInfo logger "Agent loop started"
   46   go (emptyContext (_env_systemPrompt env))
   47   where
   48     channel  = _env_channel env
   49     logger   = _env_logger env
   50     registry = _env_registry env
   51     tools    = registryDefinitions registry
   52 
   53     go ctx = do
   54       receiveResult <- try @IOException (_ch_receive channel)
   55       case receiveResult of
   56         Left _ -> _lh_logInfo logger "Session ended"
   57         Right msg
   58           | T.null stripped -> go ctx
   59           -- INVARIANT: any message beginning with '/' is handled locally and
   60           -- NEVER forwarded to the provider. Unknown slash commands get an
   61           -- error response rather than silently routing to the LLM.
   62           | "/" `T.isPrefixOf` stripped ->
   63               case parseSlashCommand stripped of
   64                 Just cmd -> do
   65                   _lh_logInfo logger $ "Slash command: " <> stripped
   66                   ctx' <- executeSlashCommand env cmd ctx
   67                   go ctx'
   68                 Nothing -> do
   69                   _lh_logWarn logger $ "Unrecognized slash command: " <> stripped
   70                   _ch_send channel
   71                     (OutgoingMessage ("Unknown command: " <> stripped
   72                       <> "\nType /status for session info, /help for available commands."))
   73                   go ctx
   74           | otherwise -> do
   75               target <- readIORef (_env_target env)
   76               case target of
   77                 TargetHarness name -> do
   78                   harnesses <- readIORef (_env_harnesses env)
   79                   case Map.lookup name harnesses of
   80                     Nothing -> do
   81                       _ch_send channel (OutgoingMessage
   82                         ("Harness \"" <> name <> "\" is not running. Use /harness start "
   83                           <> name <> " or /target to switch targets."))
   84                       go ctx
   85                     Just hh -> do
   86                       _lh_logInfo logger $ "Routing to harness: " <> name
   87                       _hh_send hh (TE.encodeUtf8 stripped)
   88                       output <- _hh_receive hh
   89                       let response = sanitizeHarnessOutput (TE.decodeUtf8 output)
   90                       unless (T.null (T.strip response)) $
   91                         _ch_send channel (OutgoingMessage response)
   92                       go ctx
   93                 TargetProvider -> do
   94                   mProvider <- readIORef (_env_provider env)
   95                   case mProvider of
   96                     Nothing -> do
   97                       _ch_send channel (OutgoingMessage noProviderMessage)
   98                       go ctx
   99                     Just provider -> do
  100                       let userMsg = textMessage User stripped
  101                           ctx' = addMessage userMsg ctx
  102                       _lh_logDebug logger $
  103                         "Sending " <> T.pack (show (length (contextMessages ctx'))) <> " messages"
  104                       -- Wrap provider with transcript logging if configured
  105                       mTranscript <- readIORef (_env_transcript env)
  106                       model <- readIORef (_env_model env)
  107                       let provider' = case mTranscript of
  108                             Just th -> mkTranscriptProvider th (unModelId model) provider
  109                             Nothing -> provider
  110                       handleCompletion provider' ctx'
  111           where stripped = T.strip (_im_content msg)
  112 
  113     handleCompletion provider ctx = do
  114       model <- readIORef (_env_model env)
  115       let req = CompletionRequest
  116             { _cr_model        = model
  117             , _cr_messages     = contextMessages ctx
  118             , _cr_systemPrompt = contextSystemPrompt ctx
  119             , _cr_maxTokens    = Just 4096
  120             , _cr_tools        = tools
  121             , _cr_toolChoice   = Nothing
  122             }
  123       responseRef <- newIORef (Nothing :: Maybe CompletionResponse)
  124       streamedRef <- newIORef False
  125       providerResult <- try @SomeException $
  126         completeStream provider req $ \case
  127           StreamText t -> do
  128             _ch_sendChunk channel (ChunkText t)
  129             writeIORef streamedRef True
  130           StreamDone resp ->
  131             writeIORef responseRef (Just resp)
  132           _ -> pure ()
  133       case providerResult of
  134         Left e -> do
  135           _lh_logError logger $ "Provider error: " <> T.pack (show e)
  136           _ch_sendError channel (TemporaryError "Something went wrong. Please try again.")
  137           go ctx
  138         Right () -> do
  139           wasStreaming <- readIORef streamedRef
  140           when wasStreaming $ _ch_sendChunk channel ChunkDone
  141           mResp <- readIORef responseRef
  142           case mResp of
  143             Nothing -> go ctx  -- shouldn't happen
  144             Just response -> do
  145               let calls = toolUseCalls response
  146                   text = responseText response
  147                   ctx' = recordUsage (_crsp_usage response)
  148                        $ addMessage (Message Assistant (_crsp_content response)) ctx
  149               -- Send the full text. For streaming channels, the text was already
  150               -- displayed chunk-by-chunk so we skip the full send to avoid duplicates.
  151               unless (wasStreaming && _ch_streaming channel || T.null (T.strip text)) $
  152                 _ch_send channel (OutgoingMessage text)
  153               -- If there are tool calls, execute them and continue
  154               if null calls
  155                 then go ctx'
  156                 else do
  157                   results <- mapM executeCall calls
  158                   let resultMsg = toolResultMessage results
  159                       ctx'' = addMessage resultMsg ctx'
  160                   _lh_logDebug logger $
  161                     "Executed " <> T.pack (show (length results)) <> " tool calls, continuing"
  162                   handleCompletion provider ctx''
  163 
  164     executeCall (callId, name, input) = do
  165       _lh_logInfo logger $ "Tool call: " <> name
  166       result <- executeTool registry name input
  167       case result of
  168         Nothing -> do
  169           _lh_logWarn logger $ "Unknown tool: " <> name
  170           pure (callId, [TRPText ("Unknown tool: " <> name)], True)
  171         Just (parts, isErr) -> do
  172           when isErr $ _lh_logWarn logger $ "Tool error in " <> name <> ": " <> partsToText parts
  173           pure (callId, parts, isErr)
  174 
  175     partsToText :: [ToolResultPart] -> Text
  176     partsToText parts = T.intercalate "\n" [t | TRPText t <- parts]
  177 
  178 -- | Message shown when user sends a chat message but no provider is configured.
  179 noProviderMessage :: Text
  180 noProviderMessage = T.intercalate "\n"
  181   [ "No provider configured. To start chatting, configure your provider with:"
  182   , ""
  183   , "  /provider <PROVIDER>"
  184   , ""
  185   ]

  187 -- | Sanitize harness output for display in a TUI.
  188 -- Strips ANSI escape sequences (CSI, OSC, DCS, etc.), C0\/C1 control
  189 -- characters, and decorative Unicode (box drawing, block elements,
  190 -- Private Use Area, etc.) that TUI applications use for rendering.
  191 -- Also trims leading and trailing blank lines from tmux capture output.
  192 sanitizeHarnessOutput :: Text -> Text
  193 sanitizeHarnessOutput =
  194     trimBlankLines . T.pack . go . T.unpack
  195   where
  196     trimBlankLines =
  197       T.intercalate "\n"
  198       . dropWhileEnd isBlankLine
  199       . dropWhile isBlankLine
  200       . T.splitOn "\n"
  201 
  202     isBlankLine = T.all Char.isSpace
  203 
  204     dropWhileEnd _ [] = []
  205     dropWhileEnd p xs = reverse (dropWhile p (reverse xs))
  206 
  207     go [] = []
  208     go ('\ESC' : rest) = skipEscape rest
  209     -- Keep newlines and tabs
  210     go ('\n' : cs) = '\n' : go cs
  211     go ('\t' : cs) = '\t' : go cs
  212     -- Replace carriage return with newline (handles \r\n and bare \r)
  213     go ('\r' : '\n' : cs) = '\n' : go cs
  214     go ('\r' : cs) = '\n' : go cs
  215     -- Drop control characters, then decorative Unicode
  216     go (c : cs)
  217       | Char.isControl c  = go cs
  218       | isDecorativeChar c = go cs
  219       | otherwise          = c : go cs
  220 
  221     -- Skip ESC [ ... (final byte) — CSI sequences
  222     skipEscape ('[' : cs) = skipCsi cs
  223     -- Skip ESC ] ... ST — OSC sequences (terminated by BEL or ESC \)
  224     skipEscape (']' : cs) = skipOsc cs
  225     -- Skip ESC P ... ST — DCS sequences
  226     skipEscape ('P' : cs) = skipOsc cs
  227     -- Skip ESC ( X, ESC ) X — charset designators
  228     skipEscape ('(' : _ : cs) = go cs
  229     skipEscape (')' : _ : cs) = go cs
  230     -- Skip ESC followed by any single character (SS2, SS3, etc.)
  231     skipEscape (_ : cs) = go cs
  232     skipEscape [] = []
  233 
  234     -- CSI: skip parameter bytes (0x30-0x3F) and intermediate bytes (0x20-0x2F)
  235     -- until a final byte (0x40-0x7E)
  236     skipCsi [] = []
  237     skipCsi (c : cs)
  238       | c >= '@' && c <= '~' = go cs  -- final byte, done
  239       | otherwise             = skipCsi cs
  240 
  241     -- OSC / DCS: skip until BEL (0x07) or ST (ESC \)
  242     skipOsc [] = []
  243     skipOsc ('\BEL' : cs) = go cs
  244     skipOsc ('\ESC' : '\\' : cs) = go cs
  245     skipOsc (_ : cs) = skipOsc cs
  246 
  247 -- | Characters used by TUI applications for rendering decorative elements.
  248 -- These are valid Unicode but produce visual garbage when displayed outside
  249 -- the originating terminal application.
  250 isDecorativeChar :: Char -> Bool
  251 isDecorativeChar c = let cp = Char.ord c in
  252   -- Box Drawing (U+2500–U+257F)
  253      (cp >= 0x2500 && cp <= 0x257F)
  254   -- Block Elements (U+2580–U+259F)
  255   || (cp >= 0x2580 && cp <= 0x259F)
  256   -- Geometric Shapes (U+25A0–U+25FF) — squares, circles, triangles
  257   || (cp >= 0x25A0 && cp <= 0x25FF)
  258   -- Braille Patterns (U+2800–U+28FF) — used for sparklines/graphs
  259   || (cp >= 0x2800 && cp <= 0x28FF)
  260   -- Private Use Area (U+E000–U+F8FF) — Powerline, Nerd Font icons
  261   || (cp >= 0xE000 && cp <= 0xF8FF)
  262   -- Supplementary Private Use Areas (U+F0000–U+10FFFF)
  263   || cp >= 0xF0000
