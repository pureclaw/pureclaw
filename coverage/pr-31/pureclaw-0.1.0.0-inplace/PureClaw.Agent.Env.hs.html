   29 -- | All runtime dependencies for the agent loop, gathered into a single record.
   30 -- This replaces the multi-parameter signature of 'runAgentLoop' and
   31 -- 'executeSlashCommand', making it easy to add new capabilities (e.g.
   32 -- 'VaultHandle') in later work units without touching call sites.
   33 data AgentEnv = AgentEnv
   34   { _env_provider     :: IORef (Maybe SomeProvider)
   35     -- ^ The LLM provider. 'Nothing' when no credentials are configured yet.
   36   , _env_model        :: IORef ModelId
   37     -- ^ The model to use for completions. Mutable so slash commands
   38     -- like @\/provider@ and @\/target@ can hot-swap it.
   39   , _env_channel      :: ChannelHandle
   40     -- ^ The channel to read messages from and write responses to.
   41   , _env_logger       :: LogHandle
   42     -- ^ Structured logger for diagnostic output.
   43   , _env_systemPrompt :: Maybe Text
   44     -- ^ Optional system prompt prepended to every conversation.
   45   , _env_registry     :: ToolRegistry
   46     -- ^ All registered tools available for the agent to call.
   47   , _env_vault        :: IORef (Maybe VaultHandle)
   48     -- ^ Optional secrets vault. 'Nothing' if no vault is configured.
   49   , _env_pluginHandle :: PluginHandle
   50     -- ^ Handle for detecting and generating age plugin identities.
   51   , _env_transcript :: IORef (Maybe TranscriptHandle)
   52     -- ^ Optional transcript handle. When 'Just', the provider is wrapped
   53     -- with 'mkTranscriptProvider' to log all completions.
   54   , _env_policy :: SecurityPolicy
   55     -- ^ Security policy for command authorization. Needed by harness management.
   56   , _env_harnesses :: IORef (Map Text HarnessHandle)
   57     -- ^ Running harness handles, keyed by name (e.g. "claude-code").
   58   , _env_target :: IORef MessageTarget
   59     -- ^ Where incoming user messages are routed. Mutable so @\/target@
   60     -- can hot-swap the destination.
   61   , _env_nextWindowIdx :: IORef Int
   62     -- ^ Monotonically increasing counter for assigning tmux window indices
   63     -- to new harnesses. Starts at 0.
   64   }
