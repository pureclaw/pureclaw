   16 -- | Status of a harness process.
   17 data HarnessStatus
   18   = HarnessRunning
   19   | HarnessExited ExitCode
   20   deriving stock (Show, Eq)
   21 
   22 -- | Errors that can occur during harness operations.
   23 data HarnessError
   24   = HarnessNotAuthorized CommandError
   25   | HarnessBinaryNotFound Text
   26   | HarnessTmuxNotAvailable Text  -- ^ detail message (stderr from tmux, or "not found")
   27   deriving stock (Show, Eq)
   28 
   29 -- | Capability handle for interacting with a harness (e.g. Claude Code in tmux).
   30 data HarnessHandle = HarnessHandle
   31   { _hh_send    :: ByteString -> IO ()   -- ^ Write to harness input
   32   , _hh_receive :: IO ByteString         -- ^ Read harness output (scrollback capture)
   33   , _hh_name    :: Text                  -- ^ Human-readable name
   34   , _hh_session :: Text                  -- ^ tmux session name
   35   , _hh_status  :: IO HarnessStatus      -- ^ Check if running
   36   , _hh_stop    :: IO ()                 -- ^ Kill and cleanup
   37   }
   38 
   39 -- | No-op harness handle for testing.
   40 mkNoOpHarnessHandle :: HarnessHandle
   41 mkNoOpHarnessHandle = HarnessHandle
   42   { _hh_send    = \_ -> pure ()
   43   , _hh_receive = pure ""
   44   , _hh_name    = ""
   45   , _hh_session = ""
   46   , _hh_status  = pure HarnessRunning
   47   , _hh_stop    = pure ()
   48   }
