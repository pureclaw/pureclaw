   70 -- | Build the authentication headers for a request.
   71 -- For OAuth, checks token expiry and refreshes if needed.
   72 buildAuthHeaders :: AnthropicProvider -> IO [Header.Header]
   73 buildAuthHeaders p = case _ap_auth p of
   74   ApiKeyAuth key ->
   75     pure [ ("x-api-key",          withApiKey key id)
   76          , ("anthropic-version",  "2023-06-01")
   77          , ("content-type",       "application/json")
   78          ]
   79   OAuthAuth oauthHandle -> do
   80     tokens <- readIORef (_oah_tokensRef oauthHandle)
   81     now    <- getCurrentTime
   82     freshTokens <-
   83       if _oat_expiresAt tokens <= now
   84         then do
   85           newT <- _oah_refresh oauthHandle (_oat_refreshToken tokens)
   86           writeIORef (_oah_tokensRef oauthHandle) newT
   87           pure newT
   88         else pure tokens
   89     pure [ ("authorization",     "Bearer " <> withBearerToken (_oat_accessToken freshTokens) id)
   90          , ("anthropic-version", "2023-06-01")
   91          , ("anthropic-beta",    "oauth-2025-04-20")
   92          , ("content-type",      "application/json")
   93          ]

  116 -- | Call the Anthropic Messages API.
  117 anthropicComplete :: AnthropicProvider -> CompletionRequest -> IO CompletionResponse
  118 anthropicComplete provider req = do
  119   authHeaders <- buildAuthHeaders provider
  120   initReq <- HTTP.parseRequest anthropicBaseUrl
  121   let httpReq = initReq
  122         { HTTP.method         = "POST"
  123         , HTTP.requestBody    = HTTP.RequestBodyLBS (encodeRequest req)
  124         , HTTP.requestHeaders = authHeaders
  125         }
  126   resp <- HTTP.httpLbs httpReq (_ap_manager provider)
  127   let status = Status.statusCode (HTTP.responseStatus resp)
  128   if status /= 200
  129     then throwIO (AnthropicAPIError status (BL.toStrict (HTTP.responseBody resp)))
  130     else case decodeResponse (HTTP.responseBody resp) of
  131       Left err -> throwIO (AnthropicParseError (T.pack err))
  132       Right response -> pure response

  134 -- | Encode a completion request as JSON for the Anthropic API.
  135 encodeRequest :: CompletionRequest -> BL.ByteString
  136 encodeRequest = encodeRequestWith False

  138 -- | Internal: shared encoder for streaming and non-streaming requests.
  139 encodeRequestWith :: Bool -> CompletionRequest -> BL.ByteString
  140 encodeRequestWith stream req = encode $ object $
  141   [ "model"      .= unModelId (_cr_model req)
  142   , "max_tokens" .= fromMaybe 4096 (_cr_maxTokens req)
  143   , "messages"   .= map encodeMsg (_cr_messages req)
  144   ]
  145   ++ ["stream" .= True | stream]
  146   ++ maybe [] (\s -> ["system" .= s]) (_cr_systemPrompt req)
  147   ++ if null (_cr_tools req)
  148      then maybe [] (\tc -> ["tool_choice" .= encodeToolChoice tc]) (_cr_toolChoice req)
  149      else ("tools" .= map encodeTool (_cr_tools req))
  150         : maybe [] (\tc -> ["tool_choice" .= encodeToolChoice tc]) (_cr_toolChoice req)

  152 encodeMsg :: Message -> Value
  153 encodeMsg msg = object
  154   [ "role"    .= roleToText (_msg_role msg)
  155   , "content" .= map encodeContentBlock (_msg_content msg)
  156   ]

  158 encodeContentBlock :: ContentBlock -> Value
  159 encodeContentBlock (TextBlock t) = object
  160   [ "type" .= ("text" :: Text)
  161   , "text" .= t
  162   ]
  163 encodeContentBlock (ToolUseBlock callId name input) = object
  164   [ "type"  .= ("tool_use" :: Text)
  165   , "id"    .= unToolCallId callId
  166   , "name"  .= name
  167   , "input" .= input
  168   ]
  169 encodeContentBlock (ImageBlock mediaType imageData) = object
  170   [ "type" .= ("image" :: Text)
  171   , "source" .= object
  172       [ "type"       .= ("base64" :: Text)
  173       , "media_type" .= mediaType
  174       , "data"       .= TE.decodeUtf8 imageData
  175       ]
  176   ]
  177 encodeContentBlock (ToolResultBlock callId parts isErr) = object $
  178   [ "type"        .= ("tool_result" :: Text)
  179   , "tool_use_id" .= unToolCallId callId
  180   , "content"     .= map encodeToolResultPart parts
  181   ]
  182   ++ ["is_error" .= True | isErr]
  183 
  184 encodeToolResultPart :: ToolResultPart -> Value
  185 encodeToolResultPart (TRPText t) = object
  186   [ "type" .= ("text" :: Text), "text" .= t ]
  187 encodeToolResultPart (TRPImage mediaType imageData) = object
  188   [ "type" .= ("image" :: Text)
  189   , "source" .= object
  190       [ "type"       .= ("base64" :: Text)
  191       , "media_type" .= mediaType
  192       , "data"       .= TE.decodeUtf8 imageData

  211 -- | Decode an Anthropic API response into a 'CompletionResponse'.
  212 decodeResponse :: BL.ByteString -> Either String CompletionResponse
  213 decodeResponse bs = eitherDecode bs >>= parseEither parseResp
  214   where
  215     parseResp :: Value -> Parser CompletionResponse
  216     parseResp = withObject "AnthropicResponse" $ \o -> do
  217       contentArr <- o .: "content"
  218       blocks <- mapM parseBlock contentArr
  219       modelText <- o .: "model"
  220       usageObj <- o .: "usage"
  221       inToks <- usageObj .: "input_tokens"
  222       outToks <- usageObj .: "output_tokens"
  223       pure CompletionResponse
  224         { _crsp_content = blocks
  225         , _crsp_model   = ModelId modelText
  226         , _crsp_usage   = Just (Usage inToks outToks)
  227         }
  228 
  229     parseBlock :: Value -> Parser ContentBlock
  230     parseBlock = withObject "ContentBlock" $ \b -> do
  231       bType <- b .: "type"
  232       case (bType :: Text) of
  233         "text" -> TextBlock <$> b .: "text"
  234         "tool_use" -> do
  235           callId <- b .: "id"
  236           name <- b .: "name"
  237           input <- b .: "input"
  238           pure (ToolUseBlock (ToolCallId callId) name input)
  239         other -> fail $ "Unknown content block type: " <> T.unpack other

  245 -- | Stream a completion from the Anthropic API.
  246 -- Processes SSE events and emits StreamEvent callbacks. Accumulates
  247 -- the full response for the final StreamDone event.
  248 anthropicCompleteStream :: AnthropicProvider -> CompletionRequest -> (StreamEvent -> IO ()) -> IO ()
  249 anthropicCompleteStream provider req callback = do
  250   authHeaders <- buildAuthHeaders provider
  251   initReq <- HTTP.parseRequest anthropicBaseUrl
  252   let httpReq = initReq
  253         { HTTP.method         = "POST"
  254         , HTTP.requestBody    = HTTP.RequestBodyLBS (encodeStreamRequest req)
  255         , HTTP.requestHeaders = authHeaders
  256         }
  257   HTTP.withResponse httpReq (_ap_manager provider) $ \resp -> do
  258     let status = Status.statusCode (HTTP.responseStatus resp)
  259     if status /= 200
  260       then do
  261         body <- BL.toStrict <$> HTTP.brReadSome (HTTP.responseBody resp) (1024 * 1024)
  262         throwIO (AnthropicAPIError status body)
  263       else
  264         let body = HTTP.responseBody resp
  265             go st buf = do
  266               chunk <- HTTP.brRead body
  267               if BS.null chunk
  268                 then do
  269                   let (final, finalEvs) = finalizeStreamState st
  270                   mapM_ callback finalEvs
  271                   callback (StreamDone final)
  272                 else do
  273                   let (lines', remaining) = splitSSELines (buf <> chunk)
  274                   st' <- foldM (processStreamLine callback) st lines'
  275                   go st' remaining
  276         in go initialStreamState BS.empty

  278 -- | Split a buffer into complete SSE lines and remaining partial data.
  279 -- Strips a trailing CR from each complete line so CRLF-delimited
  280 -- streams parse identically to LF-delimited ones.
  281 --
  282 -- The last element of 'BS.split' is everything after the final newline
  283 -- (possibly empty) — that becomes the carry-forward partial line. The
  284 -- preceding elements are complete lines.
  285 splitSSELines :: ByteString -> ([ByteString], ByteString)
  286 splitSSELines bs = go [] (BS.split 0x0a bs)
  287   where
  288     go acc []     = (reverse acc, BS.empty)
  289     go acc [tl]   = (reverse acc, tl)
  290     go acc (l:ls) = go (stripCR l : acc) ls
  291     stripCR b = case BS.unsnoc b of
  292       Just (rest, 0x0d) -> rest
  293       _                 -> b

  295 -- | Pure accumulator for streaming SSE responses. Tracks the evolving
  296 -- list of content blocks, the model id, token counts (which arrive in
  297 -- separate events), and any tool_use block currently being assembled
  298 -- (whose JSON input arrives across multiple deltas).
  299 data StreamState = StreamState
  300   { _ss_blocks       :: [ContentBlock]    -- ^ Reversed: head is most-recent block
  301   , _ss_model        :: ModelId
  302   , _ss_inputTokens  :: Maybe Int         -- ^ From @message_start@
  303   , _ss_outputTokens :: Maybe Int         -- ^ From @message_delta@
  304   , _ss_tool         :: Maybe ToolBuilder
  305   }
  306   deriving stock (Show, Eq)
  307 
  308 -- | A tool_use block being assembled mid-stream. Inputs arrive as
  309 -- a sequence of @input_json_delta@ fragments which must be concatenated
  310 -- and parsed once @content_block_stop@ closes the block.
  311 data ToolBuilder = ToolBuilder
  312   { _tb_id    :: ToolCallId
  313   , _tb_name  :: Text
  314   , _tb_input :: ByteString
  315   }
  316   deriving stock (Show, Eq)
  317 
  318 -- | Initial state at the start of a stream.
  319 initialStreamState :: StreamState
  320 initialStreamState = StreamState [] (ModelId "") Nothing Nothing Nothing
  321 
  322 -- | Pure step: apply one parsed SSE event to the state and return any
  323 -- 'StreamEvent's that should be emitted to the caller's callback.
  324 stepStreamState :: SSEEvent -> StreamState -> (StreamState, [StreamEvent])
  325 stepStreamState evt st = case evt of
  326   SSEContentText t ->
  327     -- Accumulate streamed text into a single TextBlock rather than
  328     -- creating one per chunk (which would insert spurious newlines
  329     -- when 'responseText' joins them with "\n").
  330     let blocks' = case _ss_blocks st of
  331           (TextBlock prev : rest) -> TextBlock (prev <> t) : rest
  332           _                       -> TextBlock t : _ss_blocks st
  333     in (st { _ss_blocks = blocks' }, [StreamText t])
  334   SSEToolStart callId name ->
  335     -- Defensive: close any tool that didn't get an explicit stop.
  336     let (st', mWarn) = closeOpenTool st
  337     in (st' { _ss_tool = Just (ToolBuilder callId name BS.empty) },
  338         warningEvents mWarn ++ [StreamToolUse callId name])
  339   SSEToolDelta t ->
  340     let st' = st { _ss_tool = fmap (appendToolInput t) (_ss_tool st) }
  341     in (st', [StreamToolInput t])
  342   SSEContentBlockStop ->
  343     let (st', mWarn) = closeOpenTool st
  344     in (st', warningEvents mWarn)
  345   SSEMessageStart model mInTokens ->
  346     -- 'message_start' carries an initial @output_tokens: 1@ that we
  347     -- intentionally drop: the authoritative output_tokens count arrives
  348     -- in 'message_delta' at end-of-stream, and overwriting it here would
  349     -- only confuse the accumulator.
  350     (st { _ss_model = model, _ss_inputTokens = mInTokens }, [])
  351   SSEOutputUsage outToks ->
  352     (st { _ss_outputTokens = Just outToks }, [])
  353   SSEMessageStop -> (st, [])
  354 
  355 warningEvents :: Maybe Text -> [StreamEvent]
  356 warningEvents = maybe [] (\t -> [StreamWarning t])
  357 
  358 appendToolInput :: Text -> ToolBuilder -> ToolBuilder
  359 appendToolInput t tb = tb { _tb_input = _tb_input tb <> TE.encodeUtf8 t }
  360 
  361 -- | Close the in-progress tool block (if any), parsing its accumulated
  362 -- JSON input. A malformed or empty buffer falls back to 'object []' so
  363 -- the block isn't lost — downstream tool execution will surface any
  364 -- schema mismatch as a normal tool error. The second component is a
  365 -- diagnostic warning carrying the tool name and id so callers can log
  366 -- it; otherwise a broken stream looks like a valid call with no args.
  367 closeOpenTool :: StreamState -> (StreamState, Maybe Text)
  368 closeOpenTool st = case _ss_tool st of
  369   Nothing -> (st, Nothing)
  370   Just tb ->
  371     let mParsed = decode (BL.fromStrict (_tb_input tb))
  372         input   = fromMaybe (object []) mParsed
  373         block   = ToolUseBlock (_tb_id tb) (_tb_name tb) input
  374         st'     = st { _ss_blocks = block : _ss_blocks st, _ss_tool = Nothing }
  375         mWarn   = case mParsed of
  376           Just _  -> Nothing
  377           Nothing -> Just $
  378             "Anthropic stream: malformed tool input JSON for tool "
  379               <> _tb_name tb
  380               <> " (id=" <> unToolCallId (_tb_id tb)
  381               <> "); falling back to empty object"
  382     in (st', mWarn)

  384 -- | Pure: parse a single SSE line and step the state. Non-data lines,
  385 -- unparseable JSON, and unknown event types yield no state change and
  386 -- no events.
  387 runStreamLine :: ByteString -> StreamState -> (StreamState, [StreamEvent])
  388 runStreamLine line st = case parseSSELine line of
  389   Nothing   -> (st, [])
  390   Just json -> case parseEither parseStreamEvent json of
  391     Left _    -> (st, [])
  392     Right evt -> stepStreamState evt st
  393 
  394 -- | IO wrapper around 'runStreamLine': step the state, emit any
  395 -- resulting 'StreamEvent's via the callback.
  396 processStreamLine :: (StreamEvent -> IO ()) -> StreamState -> ByteString -> IO StreamState
  397 processStreamLine cb st line = do
  398   let (st', evs) = runStreamLine line st
  399   mapM_ cb evs
  400   pure st'

  402 -- | Finalize the accumulator into a 'CompletionResponse'. Any
  403 -- still-open tool block is closed defensively (yielding an optional
  404 -- warning to be emitted before 'StreamDone'). Usage is reported only
  405 -- if at least one of input/output tokens was observed.
  406 finalizeStreamState :: StreamState -> (CompletionResponse, [StreamEvent])
  407 finalizeStreamState st0 =
  408   let (st, mWarn) = closeOpenTool st0
  409       mUsage = case (_ss_inputTokens st, _ss_outputTokens st) of
  410         (Nothing, Nothing) -> Nothing
  411         (mi, mo)           -> Just (Usage (fromMaybe 0 mi) (fromMaybe 0 mo))
  412       resp = CompletionResponse
  413         { _crsp_content = reverse (_ss_blocks st)
  414         , _crsp_model   = _ss_model st
  415         , _crsp_usage   = mUsage
  416         }
  417   in (resp, warningEvents mWarn)

  419 -- | Parse an SSE "data: ..." line into a JSON value. Tolerates a
  420 -- trailing CR for callers that didn't pre-strip CRLF.
  421 parseSSELine :: ByteString -> Maybe Value
  422 parseSSELine bs0
  423   | BS.isPrefixOf "data: " bs = decode (BL.fromStrict (BS.drop 6 bs))
  424   | otherwise                 = Nothing
  425   where
  426     bs = case BS.unsnoc bs0 of
  427       Just (rest, 0x0d) -> rest
  428       _                 -> bs0

  430 -- | Internal SSE event types.
  431 data SSEEvent
  432   = SSEContentText Text
  433   | SSEToolStart ToolCallId Text
  434   | SSEToolDelta Text
  435   | SSEContentBlockStop
  436   | SSEMessageStart ModelId (Maybe Int)  -- ^ model, input_tokens (if present)
  437   | SSEOutputUsage Int                   -- ^ output_tokens from @message_delta@
  438   | SSEMessageStop
  439   deriving stock (Show, Eq)
  440 
  441 -- | Parse a JSON SSE event.
  442 parseStreamEvent :: Value -> Parser SSEEvent
  443 parseStreamEvent = withObject "SSEEvent" $ \o -> do
  444   eventType <- o .: "type"
  445   case (eventType :: Text) of
  446     "message_start" -> do
  447       msg     <- o .: "message"
  448       model   <- msg .: "model"
  449       mUsage  <- msg .:? "usage"
  450       mInToks <- case mUsage of
  451         Nothing -> pure Nothing
  452         Just u  -> u .:? "input_tokens"
  453       pure (SSEMessageStart (ModelId model) mInToks)
  454     "content_block_delta" -> do
  455       delta <- o .: "delta"
  456       deltaType <- delta .: "type"
  457       case (deltaType :: Text) of
  458         "text_delta" -> SSEContentText <$> delta .: "text"
  459         "input_json_delta" -> SSEToolDelta <$> delta .: "partial_json"
  460         _ -> fail $ "Unknown delta type: " <> T.unpack deltaType
  461     "content_block_start" -> do
  462       block <- o .: "content_block"
  463       blockType <- block .: "type"
  464       case (blockType :: Text) of
  465         "tool_use" -> do
  466           callId <- block .: "id"
  467           name <- block .: "name"
  468           pure (SSEToolStart (ToolCallId callId) name)
  469         _ -> fail "Ignored block start"
  470     "content_block_stop" -> pure SSEContentBlockStop
  471     "message_delta" -> do
  472       usage <- o .:? "usage"
  473       case usage of
  474         Just u  -> SSEOutputUsage <$> u .: "output_tokens"
  475         Nothing -> pure SSEMessageStop
  476     "message_stop" -> pure SSEMessageStop
  477     _ -> fail $ "Unknown event type: " <> T.unpack eventType
