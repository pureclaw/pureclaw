   42 -- | Role in a conversation. System prompts are handled separately
   43 -- via 'CompletionRequest._cr_systemPrompt' rather than as messages,
   44 -- since providers differ on how they handle system content.
   45 data Role = User | Assistant
   46   deriving stock (Show, Eq, Ord)
   47 
   48 -- | A single content block within a message. Messages contain one or
   49 -- more content blocks, allowing mixed text and tool interactions.
   50 data ContentBlock
   51   = TextBlock Text
   52   | ImageBlock
   53       { _ib_mediaType :: Text       -- ^ MIME type (e.g. "image/png")
   54       , _ib_data      :: ByteString -- ^ base64-encoded image data
   55       }
   56   | ToolUseBlock
   57       { _tub_id    :: ToolCallId
   58       , _tub_name  :: Text
   59       , _tub_input :: Value
   60       }
   61   | ToolResultBlock
   62       { _trb_toolUseId :: ToolCallId
   63       , _trb_content   :: [ToolResultPart]
   64       , _trb_isError   :: Bool
   65       }
   66   deriving stock (Show, Eq)
   67 
   68 -- | Content within a tool result. Supports text and images so that
   69 -- vision tools can return image data alongside descriptions.
   70 data ToolResultPart
   71   = TRPText Text
   72   | TRPImage Text ByteString  -- ^ (mediaType, base64Data)
   73   deriving stock (Show, Eq)

   75 -- | A single message in a conversation. Content is a list of blocks
   76 -- to support tool use/result interleaving with text.
   77 data Message = Message
   78   { _msg_role    :: Role
   79   , _msg_content :: [ContentBlock]
   80   }
   81   deriving stock (Show, Eq)
   82 
   83 -- | Convert a role to its API text representation.
   84 roleToText :: Role -> Text
   85 roleToText User      = "user"
   86 roleToText Assistant  = "assistant"
   87 
   88 -- | Create a simple text message (the common case).
   89 textMessage :: Role -> Text -> Message
   90 textMessage role txt = Message role [TextBlock txt]
   91 
   92 -- | Create a tool result message (user role with tool results).
   93 toolResultMessage :: [(ToolCallId, [ToolResultPart], Bool)] -> Message
   94 toolResultMessage results = Message User
   95   [ ToolResultBlock callId content isErr
   96   | (callId, content, isErr) <- results
   97   ]

   99 -- | Extract concatenated text from a response's content blocks.
  100 responseText :: CompletionResponse -> Text
  101 responseText resp =
  102   let texts = [t | TextBlock t <- _crsp_content resp]
  103   in T.intercalate "\n" texts
  104 
  105 -- | Extract tool use calls from a response's content blocks.
  106 toolUseCalls :: CompletionResponse -> [(ToolCallId, Text, Value)]
  107 toolUseCalls resp =
  108   [ (_tub_id b, _tub_name b, _tub_input b)
  109   | b@ToolUseBlock {} <- _crsp_content resp
  110   ]

  153 -- | Events emitted during streaming completion.
  154 data StreamEvent
  155   = StreamText Text                -- ^ Partial text content
  156   | StreamToolUse ToolCallId Text  -- ^ Tool call started (id, name)
  157   | StreamToolInput Text           -- ^ Partial tool input JSON
  158   | StreamWarning Text             -- ^ Non-fatal provider-side anomaly (e.g. malformed tool input recovered via fallback). Surfaced so callers can log it.
  159   | StreamDone CompletionResponse  -- ^ Stream finished with full response
  160   deriving stock (Show, Eq)

  176 -- | Encode a ByteString as base64 text for JSON.
  177 bsToJSON :: ByteString -> Value
  178 bsToJSON = Aeson.String . TE.decodeUtf8 . B64.encode
  179 
  180 -- | Decode base64 text from JSON to a ByteString.
  181 bsFromJSON :: Value -> Aeson.Parser ByteString
  182 bsFromJSON = Aeson.withText "Base64ByteString" $ \t ->
  183   case B64.decode (TE.encodeUtf8 t) of
  184     Right bs -> pure bs
  185     Left err -> fail ("invalid base64: " <> err)

  187 instance ToJSON ToolResultPart where
  188   toJSON (TRPText t) = Aeson.object
  189     [ "type" .= ("text" :: Text), "text" .= t ]
  190   toJSON (TRPImage mt bs) = Aeson.object
  191     [ "type" .= ("image" :: Text), "media_type" .= mt, "data" .= bsToJSON bs ]
  192 
  193 instance FromJSON ToolResultPart where
  194   parseJSON = Aeson.withObject "ToolResultPart" $ \o -> do
  195     tag <- o .: "type" :: Aeson.Parser Text
  196     case tag of
  197       "text"  -> TRPText <$> o .: "text"
  198       "image" -> TRPImage <$> o .: "media_type" <*> (o .: "data" >>= bsFromJSON)
  199       _       -> fail ("unknown ToolResultPart type: " <> T.unpack tag)

  201 instance ToJSON ContentBlock where
  202   toJSON (TextBlock t) = Aeson.object
  203     [ "type" .= ("text" :: Text), "text" .= t ]
  204   toJSON (ImageBlock mt bs) = Aeson.object
  205     [ "type" .= ("image" :: Text), "media_type" .= mt, "data" .= bsToJSON bs ]
  206   toJSON (ToolUseBlock callId name input) = Aeson.object
  207     [ "type" .= ("tool_use" :: Text)
  208     , "id"   .= unToolCallId callId
  209     , "name" .= name
  210     , "input" .= input
  211     ]
  212   toJSON (ToolResultBlock callId content isErr) = Aeson.object
  213     [ "type"        .= ("tool_result" :: Text)
  214     , "tool_use_id" .= unToolCallId callId
  215     , "content"     .= content
  216     , "is_error"    .= isErr
  217     ]
  218 
  219 instance FromJSON ContentBlock where
  220   parseJSON = Aeson.withObject "ContentBlock" $ \o -> do
  221     tag <- o .: "type" :: Aeson.Parser Text
  222     case tag of
  223       "text" -> TextBlock <$> o .: "text"
  224       "image" -> ImageBlock <$> o .: "media_type" <*> (o .: "data" >>= bsFromJSON)
  225       "tool_use" -> (ToolUseBlock . ToolCallId
  226         <$> (o .: "id"))
  227         <*> o .: "name"
  228         <*> o .: "input"
  229       "tool_result" -> (ToolResultBlock . ToolCallId
  230         <$> (o .: "tool_use_id"))
  231         <*> o .: "content"
  232         <*> o .: "is_error"
  233       _ -> fail ("unknown ContentBlock type: " <> T.unpack tag)

  307 instance ToJSON StreamEvent where
  308   toJSON (StreamText t)          = Aeson.object
  309     [ "type" .= ("text" :: Text), "text" .= t ]
  310   toJSON (StreamToolUse cid name) = Aeson.object
  311     [ "type" .= ("tool_use" :: Text), "id" .= unToolCallId cid, "name" .= name ]
  312   toJSON (StreamToolInput t)     = Aeson.object
  313     [ "type" .= ("tool_input" :: Text), "input" .= t ]
  314   toJSON (StreamWarning t)       = Aeson.object
  315     [ "type" .= ("warning" :: Text), "message" .= t ]
  316   toJSON (StreamDone resp)       = Aeson.object
  317     [ "type" .= ("done" :: Text), "response" .= resp ]

  319 instance FromJSON StreamEvent where
  320   parseJSON = Aeson.withObject "StreamEvent" $ \o -> do
  321     tag <- o .: "type" :: Aeson.Parser Text
  322     case tag of
  323       "text"       -> StreamText <$> o .: "text"
  324       "tool_use"   -> (StreamToolUse . ToolCallId <$> (o .: "id")) <*> o .: "name"
  325       "tool_input" -> StreamToolInput <$> o .: "input"
  326       "warning"    -> StreamWarning <$> o .: "message"
  327       "done"       -> StreamDone <$> o .: "response"
  328       _            -> fail ("unknown StreamEvent type: " <> T.unpack tag)

  330 -- | LLM provider interface. Each provider (Anthropic, OpenAI, etc.)
  331 -- implements this typeclass.
  332 class Provider p where
  333   complete :: p -> CompletionRequest -> IO CompletionResponse
  334   -- | Stream a completion, calling the callback for each event.
  335   -- Default falls back to non-streaming 'complete'.
  336   completeStream :: p -> CompletionRequest -> (StreamEvent -> IO ()) -> IO ()
  337   completeStream p req callback = do
  338     resp <- complete p req
  339     callback (StreamDone resp)
  340   -- | List available models from the provider.
  341   -- Default returns @[]@ (no model listing support).
  342   listModels :: p -> IO [ModelId]
  343   listModels _ = pure []
  344 
  345 -- | Existential wrapper for runtime provider selection (e.g. from config).
  346 data SomeProvider where
  347   MkProvider :: Provider p => p -> SomeProvider
  348 
  349 instance Provider SomeProvider where
  350   complete (MkProvider p) = complete p
  351   completeStream (MkProvider p) = completeStream p
  352   listModels (MkProvider p) = listModels p
