   32 -- | Start the frontend static file server.
   33 runFrontend :: FrontendConfig -> IO ()
   34 runFrontend cfg = do
   35   putStrLn "PureClaw frontend server"
   36   putStrLn $ "  Serving: " <> _fsc_staticDir cfg
   37   putStrLn $ "  URL:     http://localhost:" <> show (_fsc_port cfg)
   38   Warp.run (_fsc_port cfg) (staticApp (_fsc_staticDir cfg))
   39 
   40 -- | WAI application that serves static files with SPA fallback.
   41 staticApp :: FilePath -> Application
   42 staticApp dir req respond = do
   43   let segments = pathInfo req
   44   -- Reject path traversal
   45   if any (\s -> s == ".." || s == ".") segments
   46     then respond $ responseLBS status400 [] "Invalid path"
   47     else do
   48       let relPath  = T.unpack (T.intercalate "/" segments)
   49           filePath = dir </> if null relPath then "index.html" else relPath
   50       exists <- doesFileExist filePath
   51       if exists
   52         then serveFile filePath respond
   53         else do
   54           -- SPA fallback: serve index.html for client-side routing
   55           let indexPath = dir </> "index.html"
   56           indexExists <- doesFileExist indexPath
   57           if indexExists
   58             then serveFile indexPath respond
   59             else respond $ responseLBS status404 [] "Not found"
   60 
   61 serveFile :: FilePath -> (Response -> IO ResponseReceived) -> IO ResponseReceived
   62 serveFile path respond = do
   63   contents <- LBS.readFile path
   64   let ct = mimeType (takeExtension path)
   65   respond $ responseLBS status200 [(hContentType, ct)] contents
   66 
   67 -- | Map file extensions to MIME types.
   68 mimeType :: String -> BS.ByteString
   69 mimeType ".html"  = "text/html; charset=utf-8"
   70 mimeType ".js"    = "application/javascript"
   71 mimeType ".css"   = "text/css"
   72 mimeType ".svg"   = "image/svg+xml"
   73 mimeType ".json"  = "application/json"
   74 mimeType ".png"   = "image/png"
   75 mimeType ".jpg"   = "image/jpeg"
   76 mimeType ".jpeg"  = "image/jpeg"
   77 mimeType ".ico"   = "image/x-icon"
   78 mimeType ".woff"  = "font/woff"
   79 mimeType ".woff2" = "font/woff2"
   80 mimeType ".map"   = "application/json"
   81 mimeType _        = "application/octet-stream"
