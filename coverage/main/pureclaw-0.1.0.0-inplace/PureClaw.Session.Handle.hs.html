   85 -- | Handle for the current conversation session.
   86 --
   87 -- Owns the on-disk session directory (mode @0o700@), the @session.json@
   88 -- metadata file (mode @0o600@), and the per-session transcript handle
   89 -- writing to @transcript.jsonl@ (mode @0o600@).
   90 data SessionHandle = SessionHandle
   91   { _sh_meta       :: IORef SessionMeta
   92     -- ^ Mutable session metadata. Updated by callers (e.g. to bump
   93     -- @last_active@) and persisted by '_sh_save'.
   94   , _sh_transcript :: TranscriptHandle
   95     -- ^ Transcript handle owned by the session.
   96   , _sh_dir        :: FilePath
   97     -- ^ On-disk session directory, i.e. @<baseDir>/<sessionId>/@.
   98   , _sh_save       :: IO ()
   99     -- ^ Persist '_sh_meta' to @session.json@ atomically (write to
  100     -- @session.json.tmp@ then rename).
  101   }
  102 
  103 -- | Reasons a resume attempt can fail.
  104 data ResumeError
  105   = ResumeMissingMetadata FilePath
  106     -- ^ No @session.json@ at the expected path.
  107   | ResumeCorruptedMetadata FilePath String
  108     -- ^ @session.json@ exists but does not parse as 'SessionMeta'.
  109     -- The second field is a human-readable recovery hint.
  110   deriving stock (Show, Eq)
  111 
  112 -- | Lookup result for 'resolveSessionRef'.
  113 data ResolveError
  114   = NotFound
  115   | Ambiguous [SessionId]
  116   deriving stock (Show, Eq)

  118 -- | Result of 'validateRuntime'. 'RuntimeFallback' carries the chosen
  119 -- fallback target plus a warning message suitable for logging.
  120 data ResolvedRuntime
  121   = RuntimeOk MessageTarget
  122   | RuntimeFallback MessageTarget Text
  123   deriving stock (Show, Eq)

  129 -- | Create a brand-new on-disk session handle.
  130 --
  131 -- Creates @\<baseDir\>/\<sessionId\>/@ with mode @0o700@, writes the
  132 -- initial @session.json@ (mode @0o600@) via 'saveMeta', and opens
  133 -- @transcript.jsonl@ (mode @0o600@) through 'mkFileTranscriptHandle'.
  134 mkSessionHandle :: LogHandle -> FilePath -> SessionMeta -> IO SessionHandle
  135 mkSessionHandle logger baseDir meta = do
  136   let sid  = unSessionId (_sm_id meta)
  137       dir  = baseDir </> T.unpack sid
  138       txp  = dir </> "transcript.jsonl"
  139   createDirectoryIfMissing True dir
  140   setFileMode dir 0o700
  141   metaRef <- newIORef meta
  142   saveMeta dir metaRef
  143   tx <- mkFileTranscriptHandle logger txp
  144   pure SessionHandle
  145     { _sh_meta       = metaRef
  146     , _sh_transcript = tx
  147     , _sh_dir        = dir
  148     , _sh_save       = saveMeta dir metaRef
  149     }
  150 
  151 -- | Atomically persist the metadata IORef to @\<dir\>/session.json@ by
  152 -- writing to a sibling @session.json.tmp@ with mode @0o600@ then
  153 -- 'renameFile'-ing it into place. The rename is atomic on POSIX, so a
  154 -- crash mid-write leaves the previous @session.json@ intact.
  155 saveMeta :: FilePath -> IORef SessionMeta -> IO ()
  156 saveMeta dir ref = do
  157   meta <- readIORef ref
  158   let finalP = dir </> "session.json"
  159       tmpP   = finalP <> ".tmp"
  160   LBS.writeFile tmpP (Aeson.encode meta)
  161   setFileMode tmpP 0o600
  162   renameFile tmpP finalP

  227 -- | Reopen an existing session by ID. Reads @session.json@, validates
  228 -- the JSON, and reopens @transcript.jsonl@ for append.
  229 resumeSession
  230   :: LogHandle
  231   -> FilePath
  232   -> SessionId
  233   -> IO (Either ResumeError SessionHandle)
  234 resumeSession logger baseDir sid = do
  235   let dir    = baseDir </> T.unpack (unSessionId sid)
  236       metaP  = dir </> "session.json"
  237       txP    = dir </> "transcript.jsonl"
  238   exists <- doesFileExist metaP
  239   if not exists
  240     then pure (Left (ResumeMissingMetadata metaP))
  241     else do
  242       raw <- LBS.readFile metaP
  243       case Aeson.eitherDecode' raw of
  244         Left err -> pure
  245           (Left (ResumeCorruptedMetadata metaP
  246             ("failed to parse session metadata: " <> err
  247               <> " — recovery hint: inspect or remove " <> metaP)))
  248         Right meta -> do
  249           metaRef <- newIORef (meta :: SessionMeta)
  250           tx <- mkFileTranscriptHandle logger txP
  251           pure (Right SessionHandle
  252             { _sh_meta       = metaRef
  253             , _sh_transcript = tx
  254             , _sh_dir        = dir
  255             , _sh_save       = saveMeta dir metaRef
  256             })

  262 -- | Read all @session.json@ files under @baseDir@, optionally filtering
  263 -- by agent name, sorted by @last_active@ descending, and capped at
  264 -- @limit@ entries.
  265 --
  266 -- Silently skips entries that fail to decode so that a single corrupted
  267 -- metadata file does not hide the remaining sessions from the user.
  268 listSessions :: FilePath -> Maybe AgentName -> Int -> IO [SessionMeta]
  269 listSessions baseDir mAgent limit = do
  270   exists <- doesDirectoryExist baseDir
  271   if not exists
  272     then pure []
  273     else do
  274       entries <- listDirectory baseDir
  275       metas <- fmap catMaybes' (traverse (tryLoad baseDir) entries)
  276       let filtered = case mAgent of
  277             Nothing -> metas
  278             Just a  -> filter (\m -> _sm_agent m == Just a) metas
  279           sorted = sortOn (Down . _sm_lastActive) filtered
  280       pure (take limit sorted)
  281   where
  282     catMaybes' = foldr (\mx acc -> case mx of
  283                           Just x  -> x : acc
  284                           Nothing -> acc) []

  286 -- | Attempt to load @\<baseDir\>/\<name\>/session.json@. Returns
  287 -- 'Nothing' if the path is not a directory, the metadata file is
  288 -- missing, unreadable, or malformed.
  289 tryLoad :: FilePath -> FilePath -> IO (Maybe SessionMeta)
  290 tryLoad baseDir name = do
  291   let dir   = baseDir </> name
  292       metaP = dir </> "session.json"
  293   isDir <- doesDirectoryExist dir
  294   if not isDir
  295     then pure Nothing
  296     else do
  297       exists <- doesFileExist metaP
  298       if not exists
  299         then pure Nothing
  300         else do
  301           eBytes <- try (LBS.readFile metaP) :: IO (Either IOException LBS.ByteString)
  302           case eBytes of
  303             Left _     -> pure Nothing
  304             Right raw  -> case Aeson.decode' raw of
  305               Just m  -> pure (Just m)
  306               Nothing -> pure Nothing

  308 -- | Resolve a user-supplied reference (exact ID or prefix) to a concrete
  309 -- 'SessionId'. Exact matches win outright; otherwise the match set is
  310 -- computed by 'T.isPrefixOf' over the full ID string.
  311 resolveSessionRef
  312   :: FilePath
  313   -> Text
  314   -> IO (Either ResolveError SessionId)
  315 resolveSessionRef baseDir ref = do
  316   exists <- doesDirectoryExist baseDir
  317   if not exists
  318     then pure (Left NotFound)
  319     else do
  320       entries <- listDirectory baseDir
  321       -- Keep only entries that actually have a session.json — avoids
  322       -- matching stray directories created by unrelated tests or users.
  323       valid <- filterM hasMetadata entries
  324       let ids = map (SessionId . T.pack) valid
  325       if ref `elem` map (\(SessionId t) -> t) ids
  326         then pure (Right (SessionId ref))
  327         else case filter (\(SessionId t) -> ref `T.isPrefixOf` t) ids of
  328           []  -> pure (Left NotFound)
  329           [x] -> pure (Right x)
  330           xs  -> pure (Left (Ambiguous xs))
  331   where
  332     hasMetadata name = do
  333       let metaP = baseDir </> name </> "session.json"
  334       doesFileExist metaP
  335 
  336     -- Local strict filterM to avoid pulling in Control.Monad just for this.
  337     filterM _ []     = pure []
  338     filterM p (x:xs) = do
  339       keep <- p x
  340       rest <- filterM p xs
  341       pure (if keep then x : rest else rest)

  347 -- | Validate a 'RuntimeType' against the currently-running harnesses.
  348 --
  349 -- * 'RTProvider' always resolves to @RuntimeOk TargetProvider@.
  350 -- * @'RTHarness' name@ resolves to @RuntimeOk (TargetHarness name)@ if
  351 --   a harness with that name is present in the map.
  352 -- * @'RTHarness' name@ resolves to @RuntimeFallback TargetProvider msg@
  353 --   if the harness is absent, where @msg@ explains the fallback for
  354 --   logging at warn level.
  355 validateRuntime :: Map Text HarnessHandle -> RuntimeType -> ResolvedRuntime
  356 validateRuntime _ RTProvider = RuntimeOk TargetProvider
  357 validateRuntime harnesses (RTHarness name)
  358   | Map.member name harnesses = RuntimeOk (TargetHarness name)
  359   | otherwise =
  360       let msg = "harness '" <> name <> "' is not running, falling back to provider"
  361        in RuntimeFallback TargetProvider msg
  362 
  363 -- | Resolve a resumed session's 'RuntimeType' to a concrete
  364 -- 'MessageTarget' given the currently-running harness map, logging a
  365 -- warning if the recorded runtime is no longer available.
  366 --
  367 -- Wraps 'validateRuntime': on 'RuntimeFallback' the provided warning
  368 -- message is routed to @_lh_logWarn@ and the fallback target (always
  369 -- 'TargetProvider') is returned; on 'RuntimeOk' the target is returned
  370 -- without logging.
  371 resolveResumedTarget
  372   :: LogHandle
  373   -> Map Text HarnessHandle
  374   -> RuntimeType
  375   -> IO MessageTarget
  376 resolveResumedTarget logger harnesses rt = case validateRuntime harnesses rt of
  377   RuntimeOk tgt -> pure tgt
  378   RuntimeFallback tgt warning -> do
  379     _lh_logWarn logger warning
  380     pure tgt

  386 -- | Flip '_sm_bootstrapConsumed' to 'True' in the session metadata
  387 -- 'IORef' and persist the change to @session.json@ via '_sh_save'.
  388 --
  389 -- Called by the agent loop the first time a provider response completes
  390 -- ('StreamDone'), via the one-shot callback installed on
  391 -- @_env_onFirstStreamDone@. Idempotent: re-invocations re-save the same
  392 -- metadata without error.
  393 markBootstrapConsumed :: SessionHandle -> IO ()
  394 markBootstrapConsumed sh = do
  395   atomicModifyIORef' (_sh_meta sh) $ \m ->
  396     (m { _sm_bootstrapConsumed = True }, ())
  397   _sh_save sh

  403 -- | Load a bounded window of recent 'Message' values from a session's
  404 -- transcript, oldest-first.
  405 --
  406 -- @loadRecentMessages th maxCount maxTokens@ reads every recorded
  407 -- transcript entry via '_th_query', converts each to a text 'Message'
  408 -- (mapping 'Request' → 'User' and 'Response' → 'Assistant'), keeps at
  409 -- most the last @maxCount@, then walks that window from newest back to
  410 -- oldest accumulating an estimated token budget (computed as
  411 -- @T.length payload `div` 4@ — a rough heuristic, not a true tokenizer).
  412 -- Once the budget is exhausted no further messages are added.
  413 --
  414 -- If the transcript contains a compaction boundary (an entry whose
  415 -- metadata includes the 'compactionMetadataKey'), only entries from the
  416 -- last such boundary onward are considered.  The compaction entry itself
  417 -- is included — its payload carries the summary text that replaces the
  418 -- earlier conversation.  This ensures that after a gateway restart the
  419 -- agent resumes with the compacted view rather than double-counting
  420 -- messages that were folded into the summary.
  421 --
  422 -- Returns the surviving messages in chronological (oldest-first) order
  423 -- so they can be replayed directly into the context.
  424 loadRecentMessages :: TranscriptHandle -> Int -> Int -> IO [Message]
  425 loadRecentMessages th maxCount maxTokens = do
  426   entries <- _th_query th TranscriptFilter
  427     { _tf_harness   = Nothing
  428     , _tf_model     = Nothing
  429     , _tf_direction = Nothing
  430     , _tf_timeRange = Nothing
  431     , _tf_limit     = Nothing
  432     }
  433   let -- Trim to entries from the last compaction boundary onward.
  434       postCompaction = trimToLastCompaction entries
  435       total     = length postCompaction
  436       countWin  = if total > maxCount
  437                     then drop (total - maxCount) postCompaction
  438                     else postCompaction
  439       -- Walk newest→oldest, adding to the budget; stop at the first
  440       -- entry that would push us OVER the limit. Always include at
  441       -- least one entry if the window is non-empty, so a single
  442       -- oversized message is not silently dropped.
  443       reversed  = reverse countWin
  444       budgeted  = goBudget 0 True reversed
  445       goBudget _    _     [] = []
  446       goBudget used first (e:es) =
  447         let cost  = T.length (_te_payload e) `div` 4
  448             used' = used + cost
  449         in if used' > maxTokens && not first
  450              then []
  451              else e : goBudget used' False es
  452   pure (map entryToMessage (reverse budgeted))
  453   where
  454     entryToMessage e =
  455       let role = case _te_direction e of
  456             Request  -> User
  457             Response -> Assistant
  458       in Message role [TextBlock (_te_payload e)]
  459 
  460 -- | If any entry carries the compaction metadata key, return entries
  461 -- from the last such entry onward (inclusive).  Otherwise return the
  462 -- full list unchanged.
  463 trimToLastCompaction :: [TranscriptEntry] -> [TranscriptEntry]
  464 trimToLastCompaction entries =
  465   case lastCompactionIdx of
  466     Nothing -> entries
  467     Just i  -> drop i entries
  468   where
  469     isCompaction e = Map.member compactionMetadataKey (_te_metadata e)
  470     lastCompactionIdx = go Nothing 0 entries
  471     go acc _ []     = acc
  472     go acc n (e:es)
  473       | isCompaction e = go (Just n) (n + 1) es
  474       | otherwise      = go acc      (n + 1) es
