   31 -- | Run the main agent loop. Reads messages from the channel, sends
   32 -- them to the provider (with tool definitions), handles tool call/result
   33 -- cycles, and writes responses back.
   34 --
   35 -- Slash commands (messages starting with '/') are intercepted and
   36 -- handled before being sent to the provider.
   37 --
   38 -- If no provider is configured ('Nothing' in the IORef), chat messages
   39 -- produce a helpful error directing the user to configure credentials.
   40 -- Slash commands always work regardless of provider state.
   41 --
   42 -- Exits cleanly on 'IOException' from the channel (e.g. EOF / Ctrl-D).
   43 -- Provider errors are logged and a 'PublicError' is sent to the channel.
   44 runAgentLoop :: AgentEnv -> IO ()
   45 runAgentLoop env = runAgentLoopWith env []
   46 
   47 -- | Like 'runAgentLoop' but seeds the initial 'Context' with the given
   48 -- messages (in chronological, oldest-first order). Used by the resume
   49 -- path to replay recent transcript entries so the agent has memory of
   50 -- prior turns.
   51 runAgentLoopWith :: AgentEnv -> [Message] -> IO ()
   52 runAgentLoopWith env initialMessages = do
   53   _lh_logInfo logger "Agent loop started"
   54   let ctx0 = replaceMessages initialMessages (emptyContext (_env_systemPrompt env))
   55   go ctx0
   56   where
   57     channel  = _env_channel env
   58     logger   = _env_logger env
   59     registry = _env_registry env
   60     tools    = registryDefinitions registry
   61 
   62     go ctx = do
   63       receiveResult <- try @IOException (_ch_receive channel)
   64       case receiveResult of
   65         Left _ -> _lh_logInfo logger "Session ended"
   66         Right msg
   67           | T.null stripped -> go ctx
   68           -- INVARIANT: any message beginning with '/' is handled locally and
   69           -- NEVER forwarded to the provider. Unknown slash commands get an
   70           -- error response rather than silently routing to the LLM.
   71           | "/" `T.isPrefixOf` stripped ->
   72               case parseSlashCommand stripped of
   73                 Just cmd -> do
   74                   _lh_logInfo logger $ "Slash command: " <> stripped
   75                   ctx' <- executeSlashCommand env cmd ctx
   76                   go ctx'
   77                 Nothing -> do
   78                   _lh_logWarn logger $ "Unrecognized slash command: " <> stripped
   79                   _ch_send channel
   80                     (OutgoingMessage ("Unknown command: " <> stripped
   81                       <> "\nType /status for session info, /help for available commands."))
   82                   go ctx
   83           | otherwise -> do
   84               target <- readIORef (_env_target env)
   85               case target of
   86                 TargetHarness name -> do
   87                   harnesses <- readIORef (_env_harnesses env)
   88                   case Map.lookup name harnesses of
   89                     Nothing -> do
   90                       _ch_send channel (OutgoingMessage
   91                         ("Harness \"" <> name <> "\" is not running. Use /harness start "
   92                           <> name <> " or /target to switch targets."))
   93                       go ctx
   94                     Just hh -> do
   95                       _lh_logInfo logger $ "Routing to harness: " <> name
   96                       _hh_send hh (TE.encodeUtf8 stripped)
   97                       output <- _hh_receive hh
   98                       let response = sanitizeHarnessOutput (TE.decodeUtf8 output)
   99                       unless (T.null (T.strip response)) $
  100                         _ch_send channel (OutgoingMessage (prefixHarnessOutput name response))
  101                       go ctx
  102                 TargetProvider -> do
  103                   mProvider <- readIORef (_env_provider env)
  104                   case mProvider of
  105                     Nothing -> do
  106                       _ch_send channel (OutgoingMessage noProviderMessage)
  107                       go ctx
  108                     Just provider -> do
  109                       let userMsg = textMessage User stripped
  110                           ctx' = addMessage userMsg ctx
  111                       _lh_logDebug logger $
  112                         "Sending " <> T.pack (show (length (contextMessages ctx'))) <> " messages"
  113                       -- Wrap provider with transcript logging (session owns the transcript)
  114                       th <- envTranscript env
  115                       model <- readIORef (_env_model env)
  116                       let provider' = mkTranscriptProvider th (unModelId model) provider
  117                       handleCompletion provider' ctx'
  118           where stripped = T.strip (_im_content msg)
  119 
  120     handleCompletion provider ctx = do
  121       model <- readIORef (_env_model env)
  122       let modelName = unModelId model
  123           req = CompletionRequest
  124             { _cr_model        = model
  125             , _cr_messages     = contextMessages ctx
  126             , _cr_systemPrompt = contextSystemPrompt ctx
  127             , _cr_maxTokens    = Just 4096
  128             , _cr_tools        = tools
  129             , _cr_toolChoice   = Nothing
  130             }
  131       responseRef <- newIORef (Nothing :: Maybe CompletionResponse)
  132       streamedRef <- newIORef False
  133       prefixSentRef <- newIORef False
  134       providerResult <- try @SomeException $
  135         completeStream provider req $ \case
  136           StreamText t -> do
  137             -- Emit origin prefix before the first streamed chunk
  138             prefixSent <- readIORef prefixSentRef
  139             unless prefixSent $ do
  140               _ch_sendChunk channel (ChunkText (modelName <> "> "))
  141               writeIORef prefixSentRef True
  142             _ch_sendChunk channel (ChunkText t)
  143             writeIORef streamedRef True
  144           StreamDone resp -> do
  145             writeIORef responseRef (Just resp)
  146             -- Fire and clear the one-shot "first StreamDone" callback
  147             -- atomically so concurrent StreamDone deliveries cannot
  148             -- race and invoke it twice. In production this is used to
  149             -- mark the active session's bootstrap as consumed.
  150             mAction <- atomicModifyIORef' (_env_onFirstStreamDone env)
  151                          (Nothing,)
  152             for_ mAction id
  153           StreamWarning w -> _lh_logWarn logger w
  154           _ -> pure ()
  155       case providerResult of
  156         Left e -> do
  157           _lh_logError logger $ "Provider error: " <> T.pack (show e)
  158           _ch_sendError channel (TemporaryError "Something went wrong. Please try again.")
  159           go ctx
  160         Right () -> do
  161           wasStreaming <- readIORef streamedRef
  162           when wasStreaming $ _ch_sendChunk channel ChunkDone
  163           mResp <- readIORef responseRef
  164           case mResp of
  165             Nothing -> go ctx  -- shouldn't happen
  166             Just response -> do
  167               let calls = toolUseCalls response
  168                   text = responseText response
  169                   ctx' = recordUsage (_crsp_usage response)
  170                        $ addMessage (Message Assistant (_crsp_content response)) ctx
  171               -- Send the full text. For streaming channels, the text was already
  172               -- displayed chunk-by-chunk so we skip the full send to avoid duplicates.
  173               unless (wasStreaming && _ch_streaming channel || T.null (T.strip text)) $
  174                 _ch_send channel (OutgoingMessage (prefixHarnessOutput modelName text))
  175               -- If there are tool calls, execute them and continue
  176               if null calls
  177                 then go ctx'
  178                 else do
  179                   results <- mapM executeCall calls
  180                   let resultMsg = toolResultMessage results
  181                       ctx'' = addMessage resultMsg ctx'
  182                   _lh_logDebug logger $
  183                     "Executed " <> T.pack (show (length results)) <> " tool calls, continuing"
  184                   handleCompletion provider ctx''
  185 
  186     executeCall (callId, name, input) = do
  187       _lh_logInfo logger $ "Tool call: " <> name
  188       result <- executeTool registry name input
  189       case result of
  190         Nothing -> do
  191           _lh_logWarn logger $ "Unknown tool: " <> name
  192           pure (callId, [TRPText ("Unknown tool: " <> name)], True)
  193         Just (parts, isErr) -> do
  194           when isErr $ _lh_logWarn logger $ "Tool error in " <> name <> ": " <> partsToText parts
  195           pure (callId, parts, isErr)
  196 
  197     partsToText :: [ToolResultPart] -> Text
  198     partsToText parts = T.intercalate "\n" [t | TRPText t <- parts]
  199 
  200 -- | Message shown when user sends a chat message but no provider is configured.
  201 noProviderMessage :: Text
  202 noProviderMessage = T.intercalate "\n"
  203   [ "No provider configured. To start chatting, configure your provider with:"
  204   , ""
  205   , "  /provider <PROVIDER>"
  206   , ""
