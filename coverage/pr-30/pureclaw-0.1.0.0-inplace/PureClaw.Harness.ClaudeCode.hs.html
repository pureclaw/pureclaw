   31 mkClaudeCodeHarness
   32   :: SecurityPolicy
   33   -> TranscriptHandle
   34   -> IO (Either HarnessError HarnessHandle)

   38     requireTmux

   42 -- | Testable variant with injectable dependencies.
   43 mkClaudeCodeHarnessWith
   44   :: IO (Maybe FilePath)                                                    -- ^ findExecutable "claude"
   45   -> IO (Either HarnessError ())                                            -- ^ requireTmux
   46   -> (Text -> Text -> FilePath -> [Text] -> IO (Either HarnessError ()))    -- ^ addHarnessWindow
   47   -> (Text -> IO (Either HarnessError ()))                                  -- ^ startTmuxSession
   48   -> SecurityPolicy
   49   -> TranscriptHandle
   50   -> IO (Either HarnessError HarnessHandle)
   51 mkClaudeCodeHarnessWith findClaude checkTmux addWindow startSession policy th =
   52   -- Step 1: Pre-check authorization (pure, no IO needed).
   53   -- This catches Deny autonomy and missing command allowlisting before any IO.
   54   case preAuthorize policy of
   55     Left cmdErr -> pure (Left (HarnessNotAuthorized cmdErr))
   56     Right () -> do
   57       -- Step 2: Check tmux availability
   58       tmuxResult <- checkTmux
   59       case tmuxResult of
   60         Left err -> pure (Left err)
   61         Right () -> do
   62           -- Step 3: Find claude binary
   63           mClaudePath <- findClaude
   64           case mClaudePath of
   65             Nothing -> pure (Left (HarnessBinaryNotFound "claude"))
   66             Just claudePath -> do
   67               -- Step 4: Authorize the full command path
   68               case authorize policy claudePath [] of
   69                 Left cmdErr -> pure (Left (HarnessNotAuthorized cmdErr))
   70                 Right authorizedCmd -> do
   71                   let program = getCommandProgram authorizedCmd
   72                   -- Step 5: Start tmux session (idempotent)
   73                   sessionResult <- startSession sessionName
   74                   case sessionResult of
   75                     Left err -> pure (Left err)
   76                     Right () -> do
   77                       -- Step 6: Add harness window
   78                       windowResult <- addWindow sessionName windowName program []
   79                       case windowResult of
   80                         Left err -> pure (Left err)
   81                         Right () -> do
   82                           -- Step 7: Wire up the HarnessHandle
   83                           let handle = HarnessHandle
   84                                 { _hh_send    = transcriptSend th
   85                                 , _hh_receive = transcriptReceive th
   86                                 , _hh_name    = "Claude Code"
   87                                 , _hh_session = sessionName
   88                                 , _hh_status  = checkWindowStatus th
   89                                 , _hh_stop    = stopHarnessWindow sessionName windowName
   90                                 }
   91                           pure (Right handle)
   92 
   93 -- | Pre-authorize: check that the policy would allow "claude" at all.
   94 -- This is a fast pure check before doing any IO (tmux, findExecutable, etc.).
   95 preAuthorize :: SecurityPolicy -> Either CommandError ()
   96 preAuthorize policy = case authorize policy "claude" [] of
   97   Left err -> Left err
   98   Right _  -> Right ()

  100 -- | Send input to the Claude Code window with transcript logging.
  101 transcriptSend :: TranscriptHandle -> ByteString -> IO ()
  102 transcriptSend th input = do
  103   entryId <- UUID.toText <$> UUID.nextRandom
  104   now <- getCurrentTime
  105   let entry = TranscriptEntry
  106         { _te_id            = entryId
  107         , _te_timestamp     = now
  108         , _te_harness       = Just "claude-code"
  109         , _te_model         = Nothing
  110         , _te_direction     = Request
  111         , _te_payload       = encodePayload input
  112         , _te_durationMs    = Nothing
  113         , _te_correlationId = entryId
  114         , _te_metadata      = Map.empty
  115         }
  116   _th_record th entry
  117   sendToWindow sessionName windowName input
  118 
  119 -- | Capture output from the Claude Code window with transcript logging.
  120 -- Uses withTranscript to wrap the capture call.
  121 transcriptReceive :: TranscriptHandle -> IO ByteString
  122 transcriptReceive th = do
  123   let target = sessionName <> ":" <> windowName
  124   output <- captureWindow target 300
  125   entryId <- UUID.toText <$> UUID.nextRandom
  126   now <- getCurrentTime
  127   let entry = TranscriptEntry
  128         { _te_id            = entryId
  129         , _te_timestamp     = now
  130         , _te_harness       = Just "claude-code"
  131         , _te_model         = Nothing
  132         , _te_direction     = Response
  133         , _te_payload       = encodePayload output
  134         , _te_durationMs    = Nothing
  135         , _te_correlationId = entryId
  136         , _te_metadata      = Map.empty
  137         }
  138   _th_record th entry
  139   pure output

  141 -- | Check if the Claude Code tmux window is still running.
  142 -- Uses @tmux list-windows@ to check if the window exists.
  143 checkWindowStatus :: TranscriptHandle -> IO HarnessStatus
  144 checkWindowStatus th = do
  145   mTmux <- findTmux
  146   case mTmux of
  147     Nothing -> pure (HarnessExited (ExitFailure 127))
  148     Just tmuxBin -> checkWithTmux tmuxBin th
  149 
  150 checkWithTmux :: FilePath -> TranscriptHandle -> IO HarnessStatus
  151 checkWithTmux tmuxBin th = do
  152   exitCode <- P.runProcess
  153     $ P.setStdin P.closed
  154     $ P.setStdout P.nullStream
  155     $ P.setStderr P.nullStream
  156     $ P.proc tmuxBin ["list-windows", "-t", "pureclaw", "-F", "#{window_name}"]
  157   case exitCode of
  158     ExitSuccess -> pure HarnessRunning
  159     ExitFailure code -> do
  160       -- Window or session is gone — log the event
  161       now <- getCurrentTime
  162       entryId <- UUID.toText <$> UUID.nextRandom
  163       let entry = TranscriptEntry
  164             { _te_id            = entryId
  165             , _te_timestamp     = now
  166             , _te_harness       = Just "claude-code"
  167         , _te_model         = Nothing
  168             , _te_direction     = Response
  169             , _te_payload       = ""
  170             , _te_durationMs    = Nothing
  171             , _te_correlationId = entryId
  172             , _te_metadata      = Map.fromList
  173                 [ ("event", Aeson.String "harness_exited")
  174                 , ("exit_code", Aeson.toJSON code)
  175                 ]
  176             }
  177       _th_record th entry
  178       pure (HarnessExited (ExitFailure code))
