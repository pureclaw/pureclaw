  217 -- | Parser for the top-level command.

  233   <|> (CmdTui <$> chatOptionsParser)  -- default to tui when no subcommand

  258 -- | Import an OpenClaw state directory.
  259 runImport :: ImportOptions -> Maybe FilePath -> IO ()
  260 runImport opts mPositional = do
  261   (fromDir, toDir) <- resolveImportOptions opts mPositional
  262   putStrLn $ "Importing OpenClaw state from: " <> fromDir
  263   putStrLn $ "Writing to: " <> toDir
  264   result <- importOpenClawDir fromDir toDir
  265   case result of
  266     Left err -> do
  267       putStrLn $ "Error: " <> T.unpack err
  268       exitFailure
  269     Right dir -> do
  270       let ir = _dir_configResult dir
  271           configDir = toDir </> "config"
  272       putStrLn ""
  273       putStrLn "Import complete!"
  274       putStrLn ""
  275       putStrLn "  Imported:"
  276       putStrLn $ "    Config:       " <> configDir </> "config.toml"
  277       case _ir_agentsWritten ir of
  278         [] -> pure ()
  279         agents -> do
  280           putStrLn $ "    Agents:       " <> T.unpack (T.intercalate ", " agents)
  281           mapM_ (\a -> putStrLn $ "                  " <> configDir </> "agents" </> T.unpack a </> "AGENTS.md") agents
  282       when (_dir_credentialsOk dir)
  283         $ putStrLn $ "    Credentials:  " <> toDir </> "credentials.json"
  284       case _dir_deviceId dir of
  285         Just did -> putStrLn $ "    Device ID:    " <> T.unpack did
  286         Nothing  -> pure ()
  287       case _dir_workspacePath dir of
  288         Just ws -> putStrLn $ "    Workspace:    " <> ws <> " (referenced in config)"
  289         Nothing -> pure ()
  290       when (_dir_modelsImported dir)
  291         $ putStrLn $ "    Models:       " <> toDir </> "models.json"
  292 
  293       -- Skipped items
  294       let skipped =
  295             [("Cron jobs", "PureClaw cron format not yet supported") | _dir_cronSkipped dir]
  296       if null skipped
  297         then pure ()
  298         else do
  299           putStrLn ""
  300           putStrLn "  Skipped:"
  301           mapM_ (\(item, reason) -> putStrLn $ "    " <> item <> ": " <> reason) skipped
  302 
  303       -- Extra workspaces
  304       case _dir_extraWorkspaces dir of
  305         [] -> pure ()
  306         ws -> do
  307           putStrLn ""
  308           putStrLn "  Additional workspaces found (noted in config comments):"
  309           mapM_ (\w -> putStrLn $ "    " <> w) ws
  310 
  311       -- Warnings
  312       let allWarnings = _ir_warnings ir <> _dir_warnings dir
  313       case allWarnings of
  314         [] -> pure ()
  315         ws -> do
  316           putStrLn ""
  317           putStrLn "  Warnings:"
  318           mapM_ (\w -> putStrLn $ "    " <> T.unpack w) ws
  319 
  320       putStrLn ""
  321       putStrLn "Next steps:"
  322       putStrLn "  1. Review the imported config and agent files"
  323       if _dir_credentialsOk dir
  324         then putStrLn "  2. Move credentials.json secrets into the PureClaw vault: /vault setup"
  325         else putStrLn "  2. Configure your API key: pureclaw tui --api-key <key>"
  326       putStrLn "  3. Run: pureclaw tui"

  328 -- | Run an interactive chat session.
  329 runChat :: ChatOptions -> IO ()
  330 runChat opts = do
  331   let logger = mkStderrLogHandle
  332 
  333   -- Load config file: --config flag overrides default search locations
  334   configResult <- maybe loadConfigDiag loadFileConfigDiag (_co_config opts)
  335   let fileCfg = configFileConfig configResult
  336 
  337   -- Log config loading result
  338   case configResult of
  339     ConfigLoaded path _ ->
  340       _lh_logInfo logger $ "Config: " <> T.pack path
  341     ConfigParseError path err -> do
  342       _lh_logWarn logger $ "Config file has errors: " <> T.pack path
  343       _lh_logWarn logger err
  344       _lh_logWarn logger "Using default configuration."
  345     ConfigFileNotFound path ->
  346       _lh_logWarn logger $ "Config file not found: " <> T.pack path
  347     ConfigNotFound _paths ->
  348       _lh_logInfo logger "No config file found"
  349 
  350   -- Resolve effective values: CLI flag > config file > default
  351   let effectiveProvider = fromMaybe Anthropic  (_co_provider opts <|> parseProviderMaybe (_fc_provider fileCfg))
  352       effectiveModel    = fromMaybe "claude-sonnet-4-20250514" (_co_model opts <|> fmap T.unpack (_fc_model fileCfg))
  353       effectiveMemory   = fromMaybe NoMemory    (_co_memory opts <|> parseMemoryMaybe (_fc_memory fileCfg))
  354       effectiveApiKey   = _co_apiKey opts <|> fmap T.unpack (_fc_apiKey fileCfg)
  355       effectiveSystem   = _co_system opts <|> fmap T.unpack (_fc_system fileCfg)
  356       effectiveAllow    = _co_allowCommands opts <> maybe [] (map T.unpack) (_fc_allow fileCfg)
  357       effectiveAutonomy = _co_autonomy opts
  358                       <|> parseAutonomyMaybe (_fc_autonomy fileCfg)
  359 
  360   -- Vault (opened before provider so API keys can be fetched from vault)
  361   vaultOpt <- resolveVault fileCfg (_co_noVault opts) logger
  362 
  363   -- Provider (may be Nothing if no credentials are configured yet)
  364   manager <- HTTP.newTlsManager
  365   mProvider <- if effectiveProvider == Anthropic && _co_oauth opts
  366     then Just <$> resolveAnthropicOAuth vaultOpt manager
  367     else resolveProvider effectiveProvider effectiveApiKey vaultOpt manager
  368 
  369   -- Model
  370   let model = ModelId (T.pack effectiveModel)
  371 
  372   -- System prompt: effective --system flag > SOUL.md > nothing
  373   sysPrompt <- case effectiveSystem of
  374     Just s  -> pure (Just (T.pack s))
  375     Nothing -> do
  376       let soulPath = fromMaybe "SOUL.md" (_co_soul opts)
  377       ident <- loadIdentity soulPath
  378       if ident == defaultIdentity
  379         then pure Nothing
  380         else pure (Just (identitySystemPrompt ident))
  381 
  382   -- Security policy
  383   let policy = buildPolicy effectiveAutonomy effectiveAllow
  384 
  385   -- Handles
  386   let workspace = WorkspaceRoot "."
  387       sh        = mkShellHandle logger
  388       fh        = mkFileHandle workspace
  389       nh        = mkNetworkHandle manager
  390   mh <- resolveMemory effectiveMemory
  391 
  392   -- Tool registry
  393   let registry = buildRegistry policy sh workspace fh mh nh
  394 
  395   hSetBuffering stdout LineBuffering
  396   case mProvider of
  397     Just _  -> _lh_logInfo logger $ "Provider: " <> T.pack (providerToText effectiveProvider)
  398     Nothing -> _lh_logInfo logger
  399       "No providers configured \x2014 use /provider to get started"
  400   _lh_logInfo logger $ "Model: " <> T.pack effectiveModel
  401   _lh_logInfo logger $ "Memory: " <> T.pack (memoryToText effectiveMemory)
  402   case (_sp_allowedCommands policy, _sp_autonomy policy) of
  403     (AllowAll, Full) -> do
  404       _lh_logInfo logger "Commands: allow all (unrestricted mode)"
  405       _lh_logInfo logger
  406         "\x26a0\xfe0f  Running in unrestricted mode \x2014 the agent can execute any command without approval."
  407     (_, Deny) ->
  408       _lh_logInfo logger "Commands: none (deny all)"
  409     (AllowList s, _) | Set.null s ->
  410       _lh_logInfo logger "Commands: none (deny all)"
  411     _ ->
  412       _lh_logInfo logger $ "Commands: " <> T.intercalate ", " (map T.pack effectiveAllow)
  413 
  414   -- Channel selection: CLI flag > config file > default (cli)
  415   let effectiveChannel = fromMaybe "cli"
  416         (_co_channel opts <|> fmap T.unpack (_fc_defaultChannel fileCfg))
  417 
  418   -- IORef for two-phase init: completer is built before AgentEnv exists,
  419   -- but reads the env at completion time via this ref.
  420   envRef <- newIORef Nothing
  421   slashCompleter <- buildCompleter envRef
  422 
  423   let startWithChannel :: ChannelHandle -> IO ()
  424       startWithChannel channel = do
  425         putStrLn "PureClaw 0.1.0 \x2014 Haskell-native AI agent runtime"
  426         case effectiveChannel of
  427           "cli" -> putStrLn "Type your message and press Enter. Ctrl-D to exit."
  428           _     -> putStrLn $ "Channel: " <> effectiveChannel
  429         putStrLn ""
  430         -- Create transcript handle
  431         pureclawDir <- getPureclawDir
  432         let transcriptDir = pureclawDir </> "transcripts"
  433         createDirectoryIfMissing True transcriptDir
  434         timestamp <- getCurrentTime
  435         let transcriptFile = transcriptDir
  436               </> formatTime defaultTimeLocale "%Y%m%d-%H%M%S" timestamp
  437                   <> "-" <> effectiveChannel <> ".jsonl"
  438         th <- mkFileTranscriptHandle logger transcriptFile
  439         transcriptRef <- newIORef (Just th)
  440         harnessRef  <- newIORef Map.empty
  441         vaultRef    <- newIORef vaultOpt
  442         providerRef <- newIORef mProvider
  443         modelRef    <- newIORef model
  444         let env = AgentEnv
  445               { _env_provider     = providerRef
  446               , _env_model        = modelRef
  447               , _env_channel      = channel
  448               , _env_logger       = logger
  449               , _env_systemPrompt = sysPrompt
  450               , _env_registry     = registry
  451               , _env_vault        = vaultRef
  452               , _env_pluginHandle = mkPluginHandle
  453               , _env_transcript   = transcriptRef
  454               , _env_policy       = policy
  455               , _env_harnesses    = harnessRef
  456               }
  457         -- Fill the envRef so the tab completer can access the live env
  458         writeIORef envRef (Just env)
  459         runAgentLoop env
  460 
  461   case effectiveChannel of
  462     "signal" -> do
  463       let sigCfg = resolveSignalConfig fileCfg
  464       -- Check that signal-cli is installed
  465       signalCliResult <- try @IOException $
  466         P.readProcess (P.proc "signal-cli" ["--version"])
  467       case signalCliResult of
  468         Left _ -> do
  469           _lh_logWarn logger "signal-cli is not installed or not in PATH."
  470           _lh_logWarn logger "Install it from: https://github.com/AsamK/signal-cli"
  471           _lh_logWarn logger "  brew install signal-cli    (macOS)"
  472           _lh_logWarn logger "  nix-env -i signal-cli      (NixOS)"
  473           _lh_logWarn logger "Falling back to CLI channel."
  474           mkCLIChannelHandle (Just slashCompleter) >>= startWithChannel
  475         Right _ -> do
  476           _lh_logInfo logger $ "Signal account: " <> _sc_account sigCfg
  477           transport <- mkSignalCliTransport (_sc_account sigCfg) logger
  478           withSignalChannel sigCfg transport logger startWithChannel
  479     "cli" ->
  480       mkCLIChannelHandle (Just slashCompleter) >>= startWithChannel
  481     other -> do
  482       _lh_logWarn logger $ "Unknown channel: " <> T.pack other <> ". Using CLI."
  483       mkCLIChannelHandle (Just slashCompleter) >>= startWithChannel

  521 -- | Build the tool registry with all available tools.
  522 buildRegistry :: SecurityPolicy -> ShellHandle -> WorkspaceRoot -> FileHandle -> MemoryHandle -> NetworkHandle -> ToolRegistry
  523 buildRegistry policy sh workspace fh mh nh =
  524   let reg = uncurry registerTool
  525   in reg (shellTool policy sh)
  526    $ reg (fileReadTool workspace fh)
  527    $ reg (fileWriteTool workspace fh)
  528    $ reg (gitTool policy sh)
  529    $ reg (memoryStoreTool mh)
  530    $ reg (memoryRecallTool mh)
  531    $ reg (httpRequestTool AllowAll nh)
  532      emptyRegistry

  534 -- | Build a security policy from optional autonomy level and allowed commands.
  535 --
  536 -- Behavior:
  537 --   * @Just Full@ + empty allow list → 'AllowAll' + 'Full' (unrestricted mode)
  538 --   * @Just Full@ + allow list → 'AllowList' of those commands + 'Full'
  539 --   * @Just Supervised@ + allow list → 'AllowList' + 'Supervised'
  540 --   * @Just Deny@ → 'defaultPolicy' ('Deny', empty 'AllowList')
  541 --   * @Nothing@ + empty allow list → 'defaultPolicy' (backward compat)
  542 --   * @Nothing@ + allow list → 'Full' + 'AllowList' (backward compat)
  543 buildPolicy :: Maybe AutonomyLevel -> [String] -> SecurityPolicy
  544 buildPolicy (Just Deny) _ = defaultPolicy
  545 buildPolicy (Just level) [] = SecurityPolicy
  546   { _sp_allowedCommands = AllowAll
  547   , _sp_autonomy = level
  548   }
  549 buildPolicy (Just level) cmds =
  550   let cmdNames = Set.fromList (map (CommandName . T.pack) cmds)
  551   in SecurityPolicy
  552     { _sp_allowedCommands = AllowList cmdNames
  553     , _sp_autonomy = level
  554     }
  555 buildPolicy Nothing [] = defaultPolicy
  556 buildPolicy Nothing cmds =
  557   let cmdNames = Set.fromList (map (CommandName . T.pack) cmds)
  558   in SecurityPolicy
  559     { _sp_allowedCommands = AllowList cmdNames
  560     , _sp_autonomy = Full
  561     }

  563 -- | Resolve the LLM provider from the provider type.
  564 -- Checks CLI flag first, then the vault for the API key.
  565 -- Returns 'Nothing' if no credentials are available (the agent loop
  566 -- will still start, allowing the user to configure credentials via
  567 -- slash commands like /vault setup).
  568 resolveProvider :: ProviderType -> Maybe String -> Maybe VaultHandle -> HTTP.Manager -> IO (Maybe SomeProvider)
  569 resolveProvider Anthropic keyOpt vaultOpt manager = do
  570   mApiKey <- resolveApiKey keyOpt "ANTHROPIC_API_KEY" vaultOpt
  571   case mApiKey of
  572     Just k  -> pure (Just (MkProvider (mkAnthropicProvider manager k)))
  573     Nothing -> do
  574       -- Fall back to cached OAuth tokens in the vault
  575       cachedBs <- tryVaultLookup vaultOpt oauthVaultKey
  576       case cachedBs >>= eitherToMaybe . deserializeTokens of
  577         Nothing -> pure Nothing
  578         Just tokens -> do
  579           let cfg = defaultOAuthConfig
  580           now <- getCurrentTime
  581           t <- if _oat_expiresAt tokens <= now
  582             then do
  583               putStrLn "OAuth access token expired \x2014 refreshing..."
  584               newT <- refreshOAuthToken cfg manager (_oat_refreshToken tokens)
  585               saveOAuthTokens vaultOpt newT
  586               pure newT
  587             else pure tokens
  588           handle <- mkOAuthHandle cfg manager t
  589           pure (Just (MkProvider (mkAnthropicProviderOAuth manager handle)))
  590 resolveProvider OpenAI keyOpt vaultOpt manager = do
  591   mApiKey <- resolveApiKey keyOpt "OPENAI_API_KEY" vaultOpt
  592   pure (fmap (MkProvider . mkOpenAIProvider manager) mApiKey)
  593 resolveProvider OpenRouter keyOpt vaultOpt manager = do
  594   mApiKey <- resolveApiKey keyOpt "OPENROUTER_API_KEY" vaultOpt
  595   pure (fmap (MkProvider . mkOpenRouterProvider manager) mApiKey)
  596 resolveProvider Ollama _ _ manager =
  597   pure (Just (MkProvider (mkOllamaProvider manager)))

  599 -- | Vault key used to cache OAuth tokens between sessions.
  600 oauthVaultKey :: T.Text
  601 oauthVaultKey = "ANTHROPIC_OAUTH_TOKENS"

  603 -- | Resolve an Anthropic provider via OAuth 2.0 PKCE.
  604 -- Loads cached tokens from the vault if available; runs the full browser
  605 -- flow otherwise. Refreshes expired access tokens automatically.
  606 resolveAnthropicOAuth :: Maybe VaultHandle -> HTTP.Manager -> IO SomeProvider
  607 resolveAnthropicOAuth vaultOpt manager = do
  608   let cfg = defaultOAuthConfig
  609   cachedBs <- tryVaultLookup vaultOpt oauthVaultKey
  610   tokens <- case cachedBs >>= eitherToMaybe . deserializeTokens of
  611     Just t -> do
  612       now <- getCurrentTime
  613       if _oat_expiresAt t <= now
  614         then do
  615           putStrLn "OAuth access token expired — refreshing..."
  616           newT <- refreshOAuthToken cfg manager (_oat_refreshToken t)
  617           saveOAuthTokens vaultOpt newT
  618           pure newT
  619         else pure t
  620     Nothing -> do
  621       t <- runOAuthFlow cfg manager
  622       saveOAuthTokens vaultOpt t
  623       pure t
  624   handle <- mkOAuthHandle cfg manager tokens
  625   pure (MkProvider (mkAnthropicProviderOAuth manager handle))

  627 -- | Save OAuth tokens to the vault (best-effort; logs on failure).
  628 saveOAuthTokens :: Maybe VaultHandle -> OAuthTokens -> IO ()
  629 saveOAuthTokens Nothing      _      = pure ()
  630 saveOAuthTokens (Just vh) tokens = do
  631   result <- _vh_put vh oauthVaultKey (serializeTokens tokens)
  632   case result of
  633     Left err -> putStrLn $ "Warning: could not cache OAuth tokens: " <> show err
  634     Right () -> pure ()

  638 eitherToMaybe (Left  _) = Nothing
  639 eitherToMaybe (Right a) = Just a

  651 -- | Try to look up a key from the vault. Returns 'Nothing' if the vault is
  652 -- absent, locked, or does not contain the key.

  654 tryVaultLookup Nothing   _   = pure Nothing

  658     Right bs -> pure (Just bs)

  671 -- | Open the vault if configured. Returns 'Nothing' if @--no-vault@ is set.
  672 -- When age keys are configured, uses age public-key encryption.
  673 -- Otherwise, falls back to passphrase-based encryption (works out of the box).
  674 resolveVault :: FileConfig -> Bool -> LogHandle -> IO (Maybe VaultHandle)
  675 resolveVault _ True _ = pure Nothing
  676 resolveVault fileCfg False logger =
  677   case (_fc_vault_recipient fileCfg, _fc_vault_identity fileCfg) of
  678     (Just recipient, Just identity) -> resolveAgeVault fileCfg recipient identity logger
  679     _                               -> resolvePassphraseVault fileCfg logger
  680 
  681 -- | Resolve vault using age public-key encryption (existing behaviour).
  682 resolveAgeVault :: FileConfig -> T.Text -> T.Text -> LogHandle -> IO (Maybe VaultHandle)
  683 resolveAgeVault fileCfg recipient identity logger = do
  684   encResult <- mkAgeEncryptor
  685   case encResult of
  686     Left err -> do
  687       _lh_logInfo logger $ "Vault disabled (age not available): " <> T.pack (show err)
  688       pure Nothing
  689     Right enc -> do
  690       dir <- getPureclawDir
  691       let path  = maybe (dir ++ "/vault/vault.age") T.unpack (_fc_vault_path fileCfg)
  692           mode  = parseUnlockMode (_fc_vault_unlock fileCfg)
  693           enc'  = ageVaultEncryptor enc recipient identity
  694           cfg   = VaultConfig
  695             { _vc_path    = path
  696             , _vc_keyType = inferAgeKeyType recipient
  697             , _vc_unlock  = mode
  698             }
  699       vault <- openVault cfg enc'
  700       exists <- doesFileExist path
  701       if exists
  702         then do
  703           case mode of
  704             UnlockStartup -> do
  705               result <- _vh_unlock vault
  706               case result of
  707                 Left err -> _lh_logInfo logger $
  708                   "Vault startup unlock failed (vault will be locked): " <> T.pack (show err)
  709                 Right () -> _lh_logInfo logger "Vault unlocked."
  710             _ -> pure ()
  711           pure (Just vault)
  712         else do
  713           _lh_logInfo logger "No vault found — use `/vault setup` to create one."
  714           pure Nothing
  715 
  716 -- | Resolve vault using passphrase-based encryption (default when no age keys configured).
  717 -- Prompts for passphrase on stdin at startup (if vault file exists).
  718 resolvePassphraseVault :: FileConfig -> LogHandle -> IO (Maybe VaultHandle)
  719 resolvePassphraseVault fileCfg logger = do
  720   dir <- getPureclawDir
  721   let path = maybe (dir ++ "/vault/vault.age") T.unpack (_fc_vault_path fileCfg)
  722       cfg  = VaultConfig
  723         { _vc_path    = path
  724         , _vc_keyType = "AES-256 (passphrase)"
  725         , _vc_unlock  = UnlockStartup
  726         }
  727   let getPass = do
  728         putStr "Vault passphrase: "
  729         hFlush stdout
  730         pass <- bracket_
  731           (hSetEcho stdin False)
  732           (hSetEcho stdin True >> putStrLn "")
  733           getLine
  734         pure (TE.encodeUtf8 (T.pack pass))
  735   enc <- mkPassphraseVaultEncryptor getPass
  736   vault <- openVault cfg enc
  737   exists <- doesFileExist path
  738   if exists
  739     then do
  740       result <- _vh_unlock vault
  741       case result of
  742         Left err -> _lh_logInfo logger $
  743           "Vault unlock failed: " <> T.pack (show err)
  744         Right () -> _lh_logInfo logger "Vault unlocked."
  745       pure (Just vault)
  746     else do
  747       _lh_logInfo logger "No vault found — use `/vault setup` to create one."
  748       pure Nothing

  750 -- | Infer a human-readable key type from the age recipient prefix.
  751 inferAgeKeyType :: T.Text -> T.Text
  752 inferAgeKeyType recipient
  753   | "age-plugin-yubikey" `T.isPrefixOf` recipient = "YubiKey PIV"
  754   | "age1"               `T.isPrefixOf` recipient = "X25519"
  755   | otherwise                                      = "Unknown"

  757 -- | Parse vault unlock mode from config text.
  758 parseUnlockMode :: Maybe T.Text -> UnlockMode
  759 parseUnlockMode Nothing            = UnlockOnDemand
  760 parseUnlockMode (Just t) = case t of
  761   "startup"    -> UnlockStartup
  762   "on_demand"  -> UnlockOnDemand
  763   "per_access" -> UnlockPerAccess
  764   _            -> UnlockOnDemand

  766 -- | Resolve Signal channel config from the file config.
  767 resolveSignalConfig :: FileConfig -> SignalConfig
  768 resolveSignalConfig fileCfg =
  769   let sigCfg = _fc_signal fileCfg
  770       dmPolicy = sigCfg >>= _fsc_dmPolicy
  771       allowFrom = case dmPolicy of
  772         Just "open" -> AllowAll
  773         _ -> case sigCfg >>= _fsc_allowFrom of
  774           Nothing    -> AllowAll
  775           Just []    -> AllowAll
  776           Just users -> AllowList (Set.fromList (map UserId users))
  777   in SignalConfig
  778     { _sc_account        = fromMaybe "+0000000000" (sigCfg >>= _fsc_account)
  779     , _sc_textChunkLimit = fromMaybe 6000 (sigCfg >>= _fsc_textChunkLimit)
  780     , _sc_allowFrom      = allowFrom
  781     }
