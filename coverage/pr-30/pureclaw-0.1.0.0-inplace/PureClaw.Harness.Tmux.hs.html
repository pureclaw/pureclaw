   34 -- | Resolve the absolute path to the tmux binary.
   35 -- First checks PATH via 'findExecutable', then tries common system locations.
   36 findTmux :: IO (Maybe FilePath)
   37 findTmux = do
   38   mPath <- Dir.findExecutable "tmux"
   39   case mPath of
   40     Just p  -> pure (Just p)
   41     Nothing -> findFirstExisting fallbackPaths
   42   where
   43     fallbackPaths =
   44       [ "/opt/homebrew/bin/tmux"
   45       , "/usr/local/bin/tmux"
   46       , "/usr/bin/tmux"
   47       ]
   48     findFirstExisting [] = pure Nothing
   49     findFirstExisting (p : ps) = do
   50       exists <- Dir.doesFileExist p
   51       if exists then pure (Just p) else findFirstExisting ps
   52 
   53 -- | Check if tmux is available on PATH.
   54 requireTmux :: IO (Either HarnessError ())
   55 requireTmux = do
   56   mPath <- findTmux
   57   pure $ case mPath of
   58     Nothing -> Left (HarnessTmuxNotAvailable "tmux not found on PATH or fallback locations")
   59     Just _  -> Right ()
   60 
   61 -- | Run a tmux command, capturing stderr for diagnostics.
   62 -- Returns the exit code and stderr output.
   63 runTmux :: [String] -> IO (ExitCode, ByteString)
   64 runTmux args = do
   65   mPath <- findTmux
   66   case mPath of
   67     Nothing -> pure (ExitFailure 127, "tmux not found")
   68     Just tmuxBin -> do
   69       let config = P.setStdin P.closed
   70                  $ P.setStdout P.nullStream
   71                  $ P.setStderr P.byteStringOutput
   72                  $ P.proc tmuxBin args
   73       (exitCode, _stdout, stderr) <- P.readProcess config
   74       pure (exitCode, LBS.toStrict stderr)
   75 
   76 -- | Run a tmux command silently. Returns just the exit code.
   77 runTmuxSilent :: [String] -> IO ExitCode
   78 runTmuxSilent args = fst <$> runTmux args

   80 -- | Build a stealth shell command string for launching a binary in tmux.
   81 -- Strips TMUX env vars and wraps with script(1) for a fresh PTY.
   82 --
   83 -- macOS: @script -q \/dev\/null command args...@
   84 -- Linux: @script -qc \"command args...\" \/dev\/null@
   85 stealthShellCommand :: FilePath -> [Text] -> String
   86 stealthShellCommand binary args =
   87   let argStr = unwords (map (T.unpack . shellEscape) args)
   88       fullCmd = binary <> if null args then "" else " " <> argStr
   89       scriptWrapped
   90         | Info.os == "darwin" =
   91             "script -q /dev/null " <> fullCmd
   92         | otherwise =
   93             "script -qc \"" <> escapeForShell fullCmd <> "\" /dev/null"
   94   in "env -u TMUX -u TMUX_PANE TERM=xterm-256color " <> scriptWrapped
   95 
   96 -- | Escape a string for embedding inside double quotes in a shell command.
   97 escapeForShell :: String -> String
   98 escapeForShell = concatMap go
   99   where
  100     go '"'  = "\\\""
  101     go '\\' = "\\\\"
  102     go '$'  = "\\$"
  103     go '`'  = "\\`"
  104     go c    = [c]
  105 
  106 -- | Simple shell escaping for individual arguments.
  107 -- Wraps in single quotes and escapes embedded single quotes.
  108 shellEscape :: Text -> Text
  109 shellEscape t
  110   | T.null t = "''"
  111   | T.all isSafe t = t
  112   | otherwise = "'" <> T.replace "'" "'\\''" t <> "'"
  113   where
  114     isSafe c = c `elem` (['a'..'z'] <> ['A'..'Z'] <> ['0'..'9'] <> "-_./=:@")

  116 -- | Start a tmux session with the given name if not already running.
  117 -- Creates a detached session with 300x100 dimensions.
  118 startTmuxSession :: Text -> IO (Either HarnessError ())
  119 startTmuxSession sessionName = do
  120   tmuxCheck <- requireTmux
  121   case tmuxCheck of
  122     Left err -> pure (Left err)
  123     Right () -> do
  124       exists <- sessionExists sessionName
  125       if exists
  126         then pure (Right ())
  127         else do
  128           (exitCode, stderr) <- runTmux
  129             [ "new-session", "-d"
  130             , "-s", T.unpack sessionName
  131             , "-x", "300"
  132             , "-y", "100"
  133             ]
  134           case exitCode of
  135             ExitSuccess   -> pure (Right ())
  136             ExitFailure c -> pure (Left (HarnessTmuxNotAvailable
  137               ("tmux new-session failed (exit " <> T.pack (show c) <> "): "
  138                 <> TE.decodeUtf8Lenient stderr)))
  139 
  140 -- | Check if a tmux session with the given name exists.
  141 sessionExists :: Text -> IO Bool
  142 sessionExists sessionName = do
  143   exitCode <- runTmuxSilent ["has-session", "-t", T.unpack sessionName]
  144   pure (exitCode == ExitSuccess)

  146 -- | Add a window to a tmux session for a harness.
  147 -- Uses stealth mode: env -u TMUX, script -c for fresh PTY.
  148 addHarnessWindow :: Text -> Text -> FilePath -> [Text] -> IO (Either HarnessError ())
  149 addHarnessWindow sessionName windowName binary args = do
  150   tmuxCheck <- requireTmux
  151   case tmuxCheck of
  152     Left err -> pure (Left err)
  153     Right () -> do
  154       -- Kill any existing window with this name (idempotent)
  155       _ <- runTmuxSilent
  156         [ "kill-window", "-t", T.unpack sessionName <> ":" <> T.unpack windowName ]
  157       let stealthCmd = stealthShellCommand binary args
  158           target = T.unpack sessionName
  159       -- Try new-window first; if it fails (e.g., session has only window 0),
  160       -- rename window 0 and send the command there instead
  161       (exitCode, _stderr) <- runTmux
  162         [ "new-window"
  163         , "-t", target
  164         , "-n", T.unpack windowName
  165         , stealthCmd
  166         ]
  167       case exitCode of
  168         ExitSuccess   -> pure (Right ())
  169         ExitFailure _ -> do
  170           -- Fallback: use the existing window (rename + send-keys)
  171           _ <- runTmuxSilent
  172             [ "rename-window", "-t", target, T.unpack windowName ]
  173           _ <- runTmuxSilent
  174             [ "send-keys", "-t", target <> ":" <> T.unpack windowName
  175             , stealthCmd, "Enter"
  176             ]
  177           pure (Right ())

  179 -- | Send input to a harness window.
  180 -- Small input (<= 256 bytes) uses send-keys.
  181 -- Large input (> 256 bytes) uses load-buffer + paste-buffer.
  182 sendToWindow :: Text -> Text -> ByteString -> IO ()
  183 sendToWindow sessionName windowName input
  184   | BC.length input <= 256 = sendKeysSmall sessionName windowName input
  185   | otherwise              = sendKeysLarge sessionName windowName input
  186 
  187 -- | Send small input via tmux send-keys.
  188 sendKeysSmall :: Text -> Text -> ByteString -> IO ()
  189 sendKeysSmall sessionName windowName input = do
  190   let target = T.unpack sessionName <> ":" <> T.unpack windowName
  191   _ <- runTmuxSilent ["send-keys", "-t", target, BC.unpack input, "Enter"]
  192   pure ()
  193 
  194 -- | Send large input via tmux load-buffer + paste-buffer.
  195 sendKeysLarge :: Text -> Text -> ByteString -> IO ()
  196 sendKeysLarge sessionName windowName input = do
  197   let target = T.unpack sessionName <> ":" <> T.unpack windowName
  198   Temp.withSystemTempFile "pureclaw-tmux-input" $ \tmpPath tmpHandle -> do
  199     BC.hPut tmpHandle input
  200     IO.hClose tmpHandle
  201     _ <- runTmuxSilent ["load-buffer", tmpPath]
  202     _ <- runTmuxSilent ["paste-buffer", "-t", target]
  203     _ <- runTmuxSilent ["send-keys", "-t", target, "Enter"]
  204     pure ()

  206 -- | Capture output from a harness window (scrollback, last N lines).
  207 -- Strips ANSI escape sequences from the captured output.
  208 captureWindow :: Text -> Int -> IO ByteString
  209 captureWindow sessionName lineCount = do
  210   mPath <- findTmux
  211   case mPath of
  212     Nothing -> pure ""
  213     Just tmuxBin -> do
  214       let target = T.unpack sessionName
  215           config = P.setStdin P.closed
  216                  $ P.setStdout P.byteStringOutput
  217                  $ P.setStderr P.nullStream
  218                  $ P.proc tmuxBin
  219                      [ "capture-pane", "-t", target
  220                      , "-p"
  221                      , "-S", "-" <> show lineCount
  222                      ]
  223       (exitCode, stdout, _stderr) <- P.readProcess config
  224       case exitCode of
  225         ExitSuccess   -> pure (stripAnsi (LBS.toStrict stdout))
  226         ExitFailure _ -> pure ""
  227 
  228 -- | Strip ANSI escape sequences from a ByteString.
  229 -- Matches ESC [ ... (letter or @) sequences.
  230 stripAnsi :: ByteString -> ByteString
  231 stripAnsi = go
  232   where
  233     go input
  234       | BC.null input = BC.empty
  235       | otherwise =
  236           let (before, rest) = BC.break (== '\ESC') input
  237           in if BC.null rest
  238              then before
  239              else before <> skipEsc (BC.drop 1 rest)
  240     skipEsc input
  241       | BC.null input = BC.empty
  242       | BC.head input == '[' = skipCsi (BC.drop 1 input)
  243       | otherwise = go (BC.drop 1 input)
  244     skipCsi input
  245       | BC.null input = BC.empty
  246       | let c = BC.head input
  247       , c >= '@' && c <= '~' = go (BC.drop 1 input)
  248       | otherwise = skipCsi (BC.drop 1 input)

  250 -- | Display text in a harness window (for tee-style mirroring).
  251 -- Uses send-keys to echo the text.
  252 tmuxDisplay :: Text -> ByteString -> IO ()
  253 tmuxDisplay sessionName content = do
  254   let target = T.unpack sessionName
  255   _ <- runTmuxSilent ["send-keys", "-t", target, BC.unpack content, ""]
  256   pure ()
