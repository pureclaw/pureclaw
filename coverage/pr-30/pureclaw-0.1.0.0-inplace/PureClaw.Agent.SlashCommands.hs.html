   85 -- | Specification for a single slash command.
   86 -- 'allCommandSpecs' is the single source of truth: 'parseSlashCommand'
   87 -- is derived from '_cs_parse' and '/help' renders from '_cs_syntax' /
   88 -- '_cs_description', so the two cannot diverge.
   89 data CommandSpec = CommandSpec
   90   { _cs_syntax      :: Text          -- ^ Display syntax, e.g. "/vault add <name>"
   91   , _cs_description :: Text          -- ^ One-line description shown in '/help'
   92   , _cs_group       :: CommandGroup  -- ^ Organisational group
   93   , _cs_parse       :: Text -> Maybe SlashCommand
   94     -- ^ Try to parse a stripped, original-case input as this command.
   95     -- Match is case-insensitive on keywords; argument case is preserved.
   96   }

  229 -- | Parse a user message as a slash command.
  230 -- Implemented as 'asum' over '_cs_parse' from 'allCommandSpecs', followed
  231 -- by a catch-all for unrecognised @\/vault@ subcommands.
  232 -- Returns 'Nothing' only for input that does not begin with @\/@.
  233 parseSlashCommand :: Text -> Maybe SlashCommand
  234 parseSlashCommand input =
  235   let stripped = T.strip input
  236   in if "/" `T.isPrefixOf` stripped
  237      then asum (map (`_cs_parse` stripped) allCommandSpecs)
  238             <|> channelUnknownFallback stripped
  239             <|> vaultUnknownFallback stripped
  240             <|> transcriptUnknownFallback stripped
  241             <|> harnessUnknownFallback stripped
  242      else Nothing
  243 
  244 -- | Exact case-insensitive match.
  245 exactP :: Text -> SlashCommand -> Text -> Maybe SlashCommand
  246 exactP keyword cmd t = if T.toLower t == keyword then Just cmd else Nothing
  247 
  248 -- | Case-insensitive match for "/vault <sub>" with no argument.
  249 vaultExactP :: Text -> VaultSubCommand -> Text -> Maybe SlashCommand
  250 vaultExactP sub cmd t =
  251   if T.toLower t == "/vault " <> sub then Just (CmdVault cmd) else Nothing
  252 
  253 -- | Case-insensitive prefix match for "/vault <sub> [arg]".
  254 -- Argument is extracted from the original-case input, preserving its case.
  255 vaultArgP :: Text -> (Text -> VaultSubCommand) -> Text -> Maybe SlashCommand
  256 vaultArgP sub mkCmd t =
  257   let pfx   = "/vault " <> sub
  258       lower = T.toLower t
  259   in if lower == pfx || (pfx <> " ") `T.isPrefixOf` lower
  260      then Just (CmdVault (mkCmd (T.strip (T.drop (T.length pfx) t))))
  261      else Nothing

  263 -- | Case-insensitive match for "/provider [name]".
  264 -- With no argument, returns the list command. With an argument, returns
  265 -- the configure command with the argument preserved in original case.
  266 providerArgP :: ProviderSubCommand -> (Text -> ProviderSubCommand) -> Text -> Maybe SlashCommand
  267 providerArgP listCmd mkCfgCmd t =
  268   let pfx   = "/provider"
  269       lower = T.toLower t
  270   in if lower == pfx
  271      then Just (CmdProvider listCmd)
  272      else if (pfx <> " ") `T.isPrefixOf` lower
  273           then let arg = T.strip (T.drop (T.length pfx) t)
  274                in if T.null arg
  275                   then Just (CmdProvider listCmd)
  276                   else Just (CmdProvider (mkCfgCmd arg))
  277           else Nothing

  407 -- | Execute a slash command. Returns the (possibly updated) context.
  408 executeSlashCommand :: AgentEnv -> SlashCommand -> Context -> IO Context

  410 executeSlashCommand env CmdHelp ctx = do
  411   _ch_send (_env_channel env) (OutgoingMessage (renderHelpText allCommandSpecs))
  412   pure ctx
  413 
  414 executeSlashCommand env CmdNew ctx = do
  415   _ch_send (_env_channel env) (OutgoingMessage "Session cleared. Starting fresh.")
  416   pure (clearMessages ctx)
  417 
  418 executeSlashCommand env CmdReset _ctx = do
  419   _ch_send (_env_channel env) (OutgoingMessage "Full reset. Context and usage cleared.")
  420   pure (emptyContext (contextSystemPrompt _ctx))
  421 
  422 executeSlashCommand env CmdStatus ctx = do
  423   model <- readIORef (_env_model env)
  424   mProvider <- readIORef (_env_provider env)
  425   mVault <- readIORef (_env_vault env)
  426   mTranscript <- readIORef (_env_transcript env)
  427   harnesses <- readIORef (_env_harnesses env)
  428   let providerLine = case mProvider of
  429         Nothing -> "  Provider:  (not configured)"
  430         Just _  -> "  Provider:  configured"
  431       vaultLine = case mVault of
  432         Nothing -> "  Vault:     (not configured)"
  433         Just _  -> "  Vault:     configured"
  434       transcriptLine = case mTranscript of
  435         Nothing -> "  Transcript: disabled"
  436         Just _  -> "  Transcript: enabled"
  437       harnessLine = if Map.null harnesses
  438         then "  Harnesses: (none)"
  439         else "  Harnesses: " <> T.intercalate ", "
  440                [n <> " (" <> _hh_name h <> ")" | (n, h) <- Map.toList harnesses]
  441       policyLine = "  Policy:    " <> T.pack (show (_sp_autonomy (_env_policy env)))
  442       status = T.intercalate "\n"
  443         [ "Session status:"
  444         , "  Model:     " <> unModelId model
  445         , providerLine
  446         , policyLine
  447         , vaultLine
  448         , transcriptLine
  449         , harnessLine
  450         , ""
  451         , "  Messages:            " <> T.pack (show (contextMessageCount ctx))
  452         , "  Est. context tokens: " <> T.pack (show (contextTokenEstimate ctx))
  453         , "  Total input tokens:  " <> T.pack (show (contextTotalInputTokens ctx))
  454         , "  Total output tokens: " <> T.pack (show (contextTotalOutputTokens ctx))
  455         ]
  456   _ch_send (_env_channel env) (OutgoingMessage status)
  457   pure ctx
  458 
  459 executeSlashCommand env (CmdModel Nothing) ctx = do
  460   model <- readIORef (_env_model env)
  461   _ch_send (_env_channel env) (OutgoingMessage ("Current model: " <> unModelId model))
  462   pure ctx
  463 
  464 executeSlashCommand env (CmdModel (Just name)) ctx = do
  465   let send = _ch_send (_env_channel env) . OutgoingMessage
  466   writeIORef (_env_model env) (ModelId name)
  467   -- Persist to config.toml
  468   pureclawDir <- getPureclawDir
  469   let configPath = pureclawDir </> "config.toml"
  470   existing <- loadFileConfig configPath
  471   Dir.createDirectoryIfMissing True pureclawDir
  472   writeFileConfig configPath (existing { _fc_model = Just name })
  473   send $ "Model switched to: " <> name
  474   pure ctx
  475 
  476 executeSlashCommand env CmdModelList ctx = do
  477   let send = _ch_send (_env_channel env) . OutgoingMessage
  478   mProvider <- readIORef (_env_provider env)
  479   case mProvider of
  480     Nothing -> do
  481       send "No provider configured. Use /provider to set one up first."
  482       pure ctx
  483     Just provider -> do
  484       models <- listModels provider
  485       if null models
  486         then send "No models available (provider may not support model listing, or is unreachable)."
  487         else do
  488           let formatted = T.intercalate "\n" (map (\m -> "  " <> unModelId m) models)
  489           send ("Available models:\n" <> formatted)
  490       pure ctx
  491 
  492 executeSlashCommand env CmdCompact ctx = do
  493   mProvider <- readIORef (_env_provider env)
  494   case mProvider of
  495     Nothing -> do
  496       _ch_send (_env_channel env) (OutgoingMessage "Cannot compact: no provider configured.")
  497       pure ctx
  498     Just provider -> do
  499       model <- readIORef (_env_model env)
  500       (ctx', result) <- compactContext
  501         provider
  502         model
  503         0
  504         defaultKeepRecent
  505         ctx
  506       let msg = case result of
  507             NotNeeded         -> "Nothing to compact (too few messages)."
  508             Compacted o n     -> "Compacted: " <> T.pack (show o)
  509                               <> " messages \x2192 " <> T.pack (show n)
  510             CompactionError e -> "Compaction failed: " <> e
  511       _ch_send (_env_channel env) (OutgoingMessage msg)
  512       pure ctx'
  513 
  514 executeSlashCommand env (CmdProvider sub) ctx = do
  515   vaultOpt <- readIORef (_env_vault env)
  516   case vaultOpt of
  517     Nothing -> do
  518       _ch_send (_env_channel env) (OutgoingMessage
  519         "Vault not configured. Run /vault setup first to store provider credentials.")
  520       pure ctx
  521     Just vault ->
  522       executeProviderCommand env vault sub ctx
  523 
  524 executeSlashCommand env (CmdChannel sub) ctx = do
  525   executeChannelCommand env sub ctx
  526 
  527 executeSlashCommand env (CmdTranscript sub) ctx = do
  528   executeTranscriptCommand env sub ctx
  529 
  530 executeSlashCommand env (CmdHarness sub) ctx = do
  531   executeHarnessCommand env sub ctx
  532 
  533 executeSlashCommand env (CmdVault sub) ctx = do
  534   vaultOpt <- readIORef (_env_vault env)
  535   case sub of
  536     VaultSetup -> do
  537       executeVaultSetup env ctx
  538     _ -> case vaultOpt of
  539       Nothing -> do

  541           "No vault configured. Run /vault setup to create one.")
  542         pure ctx
  543       Just vault ->

  700   let ch   = _env_channel env
  701       send = _ch_send ch . OutgoingMessage
  702   case sub of
  703     VaultSetup ->
  704       -- VaultSetup is handled before dispatch; should not reach here.
  705       send "Use /vault setup to set up or rekey the vault."

  708     VaultAdd name -> do
  709       valueResult <- try @IOError (_ch_promptSecret ch ("Value for '" <> name <> "': "))
  710       case valueResult of
  711         Left e ->
  712           send ("Error reading secret: " <> T.pack (show e))
  713         Right value -> do
  714           result <- _vh_put vault name (TE.encodeUtf8 value)
  715           case result of
  716             Left err -> send ("Error storing secret: " <> T.pack (show err))
  717             Right () -> send ("Secret '" <> name <> "' stored.")
  718       pure ctx
  719 
  720     VaultList -> do
  721       result <- _vh_list vault
  722       case result of
  723         Left err  -> send ("Error: " <> T.pack (show err))

  725         Right names ->
  726           send ("Secrets:\n" <> T.unlines (map ("  \x2022 " <>) names))
  727       pure ctx
  728 
  729     VaultDelete name -> do
  730       confirm <- _ch_prompt ch ("Delete secret '" <> name <> "'? [y/N]: ")
  731       if T.strip confirm == "y" || T.strip confirm == "Y"
  732         then do
  733           result <- _vh_delete vault name
  734           case result of
  735             Left err -> send ("Error: " <> T.pack (show err))
  736             Right () -> send ("Secret '" <> name <> "' deleted.")

  738       pure ctx

  741       _vh_lock vault
  742       send "Vault locked."
  743       pure ctx
  744

  747       case result of
  748         Left err -> send ("Error unlocking vault: " <> T.pack (show err))
  749         Right () -> send "Vault unlocked."
  750       pure ctx
  751 
  752     VaultStatus' -> do
  753       status <- _vh_status vault
  754       let lockedText = if _vs_locked status then "Locked" else "Unlocked"
  755           msg = T.intercalate "\n"

  759             , "  Key:     " <> _vs_keyType status
  760             ]
  761       send msg
  762       pure ctx
  763 
  764     VaultUnknown unknownSub
  765       | T.null unknownSub -> do
  766           -- Bare /vault: show status + available subcommands
  767           mVault <- readIORef (_env_vault env)
  768           let vaultStatus = case mVault of
  769                 Nothing -> "Vault: not configured"

  780                 , "  /vault unlock       — Unlock the vault"
  781                 , "  /vault status       — Show vault state and key type"
  782                 ]

  785       | otherwise ->
  786           send ("Unknown vault command: " <> unknownSub <> ". Type /vault to see available commands.")

  793 -- | Interactive vault setup wizard. Detects auth mechanisms, lets the user
  794 -- choose, then creates or rekeys the vault.
  795 executeVaultSetup :: AgentEnv -> Context -> IO Context
  796 executeVaultSetup env ctx = do
  797   let ch   = _env_channel env
  798       send = _ch_send ch . OutgoingMessage
  799       ph   = _env_pluginHandle env
  800 
  801   -- Step 1: Detect available plugins
  802   plugins <- _ph_detect ph
  803 
  804   -- Step 2: Build choice menu
  805   let options = buildSetupOptions plugins
  806       menu    = formatSetupMenu options
  807   send menu
  808 
  809   -- Step 3: Read user's choice
  810   choiceText <- _ch_prompt ch "Choice: "
  811   case parseChoice (length options) (T.strip choiceText) of
  812     Nothing -> do
  813       send "Invalid choice. Setup cancelled."
  814       pure ctx
  815     Just idx -> do
  816       let chosen = snd (options !! idx)
  817       -- Step 4: Create encryptor based on choice
  818       encResult <- createEncryptorForChoice ch ph chosen
  819       case encResult of
  820         Left err -> do
  821           send err
  822           pure ctx
  823         Right (newEnc, keyLabel, mRecipient, mIdentity) -> do
  824           -- Step 5: Init or rekey
  825           vaultOpt <- readIORef (_env_vault env)
  826           case vaultOpt of
  827             Nothing -> do
  828               -- No vault handle at all: create from scratch
  829               setupResult <- firstTimeSetup env newEnc keyLabel
  830               case setupResult of
  831                 Left err -> send err
  832                 Right () -> do
  833                   send ("Vault created with " <> keyLabel <> " encryption.")
  834                   updateConfigAfterSetup mRecipient mIdentity keyLabel
  835             Just vault -> do
  836               -- Vault handle exists — but the file may not.
  837               -- Try init: if it succeeds, this is first-time setup.
  838               -- If VaultAlreadyExists, we need to rekey.
  839               initResult <- _vh_init vault
  840               case initResult of
  841                 Right () -> do
  842                   -- First-time init succeeded (file didn't exist)
  843                   send ("Vault created with " <> keyLabel <> " encryption.")
  844                   updateConfigAfterSetup mRecipient mIdentity keyLabel
  845                 Left VaultAlreadyExists -> do
  846                   -- Vault exists — rekey it
  847                   let confirmFn msg = do
  848                         send msg
  849                         answer <- _ch_prompt ch "Proceed? [y/N]: "
  850                         pure (T.strip answer == "y" || T.strip answer == "Y")
  851                   rekeyResult <- _vh_rekey vault newEnc keyLabel confirmFn
  852                   case rekeyResult of
  853                     Left (VaultCorrupted "rekey cancelled by user") ->
  854                       send "Rekey cancelled."
  855                     Left err ->
  856                       send ("Rekey failed: " <> T.pack (show err))
  857                     Right () -> do
  858                       send ("Vault rekeyed to " <> keyLabel <> ".")
  859                       updateConfigAfterSetup mRecipient mIdentity keyLabel
  860                 Left err ->
  861                   send ("Vault init failed: " <> T.pack (show err))
  862           pure ctx

  893 -- | Create an encryptor based on the user's setup choice.
  894 -- Returns (encryptor, key label, maybe recipient, maybe identity path).
  895 createEncryptorForChoice
  896   :: ChannelHandle
  897   -> PluginHandle
  898   -> SetupOption
  899   -> IO (Either Text (VaultEncryptor, Text, Maybe Text, Maybe Text))
  900 createEncryptorForChoice ch _ph SetupPassphrase = do
  901   passResult <- try @IOError (_ch_promptSecret ch "Passphrase: ")
  902   case passResult of
  903     Left e ->
  904       pure (Left ("Error reading passphrase: " <> T.pack (show e)))
  905     Right passphrase -> do
  906       enc <- mkPassphraseVaultEncryptor (pure (TE.encodeUtf8 passphrase))
  907       pure (Right (enc, "passphrase", Nothing, Nothing))
  908 createEncryptorForChoice ch _ph (SetupPlugin plugin) = do
  909   pureclawDir <- getPureclawDir
  910   let vaultDir      = pureclawDir </> "vault"
  911       identityFile  = vaultDir </> T.unpack (_ap_name plugin) <> "-identity.txt"
  912       identityFileT = T.pack identityFile
  913       cmd = T.pack (_ap_binary plugin) <> " --generate --pin-policy never --touch-policy never > " <> identityFileT
  914   Dir.createDirectoryIfMissing True vaultDir
  915   _ch_send ch (OutgoingMessage (T.intercalate "\n"
  916     [ "Run this in another terminal to generate a " <> _ap_label plugin <> " identity:"
  917     , ""
  918     , "  " <> cmd
  919     , ""
  920     , "The plugin will prompt you for a PIN and touch confirmation."
  921     , "Press Enter here when done (or 'q' to cancel)."
  922     ]))
  923   answer <- T.strip <$> _ch_prompt ch ""
  924   if answer == "q" || answer == "Q"
  925     then pure (Left "Setup cancelled.")
  926     else do
  927       exists <- Dir.doesFileExist identityFile
  928       if not exists
  929         then pure (Left ("Identity file not found: " <> identityFileT))
  930         else do
  931           contents <- TIO.readFile identityFile
  932           let outputLines = T.lines contents
  933               -- age-plugin-yubikey uses "#    Recipient: age1..."
  934               -- other plugins may use "# public key: age1..."
  935               findRecipient = L.find (\l ->
  936                 let stripped = T.strip (T.dropWhile (== '#') (T.strip l))
  937                 in T.isPrefixOf "Recipient:" stripped
  938                    || T.isPrefixOf "public key:" stripped) outputLines
  939           case findRecipient of
  940             Nothing ->
  941               pure (Left "No recipient found in identity file. Expected a '# Recipient: age1...' line.")
  942             Just rLine -> do
  943               -- Extract value after the label (Recipient: or public key:)
  944               let afterHash = T.strip (T.dropWhile (== '#') (T.strip rLine))
  945                   recipient = T.strip (T.drop 1 (T.dropWhile (/= ':') afterHash))
  946               ageResult <- mkAgeEncryptor
  947               case ageResult of
  948                 Left err ->
  949                   pure (Left ("age error: " <> T.pack (show err)))
  950                 Right ageEnc -> do
  951                   let enc = ageVaultEncryptor ageEnc recipient identityFileT
  952                   pure (Right (enc, _ap_label plugin, Just recipient, Just identityFileT))

 1036 executeTranscriptCommand :: AgentEnv -> TranscriptSubCommand -> Context -> IO Context
 1037 executeTranscriptCommand env sub ctx = do
 1038   let send = _ch_send (_env_channel env) . OutgoingMessage
 1039   mTh <- readIORef (_env_transcript env)
 1040   case mTh of
 1041     Nothing -> do
 1042       send "No transcript configured. Start with --transcript to enable logging."
 1043       pure ctx
 1044     Just th -> case sub of
 1045       TranscriptRecent mN -> do
 1046         let n = Data.Maybe.fromMaybe 20 mN
 1047             tf = emptyFilter { _tf_limit = Just n }
 1048         entries <- _th_query th tf
 1049         if null entries
 1050           then send "No entries found."
 1051           else send (T.intercalate "\n" (map formatEntry entries))
 1052         pure ctx
 1053 
 1054       TranscriptSearch query -> do
 1055         -- Search matches either harness or model name
 1056         allEntries <- _th_query th emptyFilter
 1057         let matches e = _te_harness e == Just query || _te_model e == Just query
 1058             entries = filter matches allEntries
 1059         if null entries
 1060           then send ("No entries found matching: " <> query)
 1061           else send (T.intercalate "\n" (map formatEntry entries))
 1062         pure ctx
 1063 
 1064       TranscriptPath -> do
 1065         path <- _th_getPath th
 1066         send (T.pack path)
 1067         pure ctx
 1068 
 1069       TranscriptUnknown subcmd -> do
 1070         send ("Unknown transcript command: " <> subcmd <> ". Try /help for available commands.")
 1071         pure ctx

 1073 -- | Format a transcript entry as a one-line summary.
 1074 -- Example: "[2026-04-04T15:30:00Z] ollama/llama3 Request (42ms)"
 1075 formatEntry :: TranscriptEntry -> Text
 1076 formatEntry entry =
 1077   let ts   = T.pack (show (_te_timestamp entry))
 1078       endpoint = case (_te_harness entry, _te_model entry) of
 1079         (Just h, Just m)  -> h <> "/" <> m
 1080         (Just h, Nothing) -> h
 1081         (Nothing, Just m) -> m
 1082         (Nothing, Nothing) -> "unknown"
 1083       dir  = T.pack (show (_te_direction entry))
 1084       dur  = case _te_durationMs entry of
 1085                Just ms -> " (" <> T.pack (show ms) <> "ms)"
 1086                Nothing -> ""
 1087   in "[" <> ts <> "] " <> endpoint <> " " <> dir <> dur

 1271 -- | Link to an existing Signal account by scanning a QR code.
 1272 -- signal-cli link outputs the sgnl:// URI, then blocks until the user
 1273 -- scans it. We need to stream the output to show the URI immediately.
 1274 signalLinkFlow :: AgentEnv -> Context -> IO Context
 1275 signalLinkFlow env ctx = do
 1276   let ch   = _env_channel env
 1277       send = _ch_send ch . OutgoingMessage
 1278 
 1279   send "Generating link... (this may take a moment)"
 1280 
 1281   let procConfig = P.setStdout P.createPipe
 1282                  $ P.setStderr P.createPipe
 1283                  $ P.proc "signal-cli" ["link", "-n", "PureClaw"]
 1284   startResult <- try @IOException $ P.startProcess procConfig
 1285   case startResult of
 1286     Left err -> do
 1287       send $ "Failed to start signal-cli: " <> T.pack (show err)
 1288       pure ctx
 1289     Right process -> do
 1290       let stdoutH = P.getStdout process
 1291           stderrH = P.getStderr process
 1292       -- signal-cli outputs the URI to stderr, then blocks waiting for scan.
 1293       -- Read stderr lines until we find the sgnl:// URI.
 1294       linkUri <- readUntilLink stderrH stdoutH
 1295       case linkUri of
 1296         Nothing -> do
 1297           -- Process may have exited with error
 1298           exitCode <- P.waitExitCode process
 1299           send $ "signal-cli link failed (exit " <> T.pack (show exitCode) <> ")"
 1300           pure ctx
 1301         Just uri -> do
 1302           send $ T.intercalate "\n"
 1303             [ "Open Signal on your phone:"
 1304             , "  Settings \x2192 Linked Devices \x2192 Link New Device"
 1305             , ""
 1306             , "Scan this link (or paste into a QR code generator):"
 1307             , ""
 1308             , "  " <> uri
 1309             , ""
 1310             , "Waiting for you to scan... (this will complete automatically)"
 1311             ]
 1312           -- Now wait for signal-cli to finish (user scans the code)
 1313           exitCode <- P.waitExitCode process
 1314           case exitCode of
 1315             ExitSuccess -> do
 1316               send "Linked successfully!"
 1317               detectAndWriteSignalConfig env ctx
 1318             _ -> do
 1319               send "Link failed or was cancelled."
 1320               pure ctx
 1321   where
 1322     -- Read lines from both handles looking for a sgnl:// URI.
 1323     -- signal-cli typically puts it on stderr.
 1324     readUntilLink :: Handle -> Handle -> IO (Maybe Text)
 1325     readUntilLink stderrH stdoutH = go (50 :: Int)  -- max 50 lines to prevent infinite loop
 1326       where
 1327         go 0 = pure Nothing
 1328         go n = do
 1329           lineResult <- try @IOException (hGetLine stderrH)
 1330           case lineResult of
 1331             Left _ -> do
 1332               -- stderr closed, try stdout
 1333               outResult <- try @IOException (hGetLine stdoutH)
 1334               case outResult of
 1335                 Left _    -> pure Nothing
 1336                 Right line ->
 1337                   let t = T.pack line
 1338                   in if "sgnl://" `T.isInfixOf` t
 1339                      then pure (Just (T.strip t))
 1340                      else go (n - 1)
 1341             Right line ->
 1342               let t = T.pack line
 1343               in if "sgnl://" `T.isInfixOf` t
 1344                  then pure (Just (T.strip t))
 1345                  else go (n - 1)
