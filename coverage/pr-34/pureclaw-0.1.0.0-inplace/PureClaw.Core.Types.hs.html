   61 -- | Opaque session identifier. The string format is produced by
   62 -- 'PureClaw.Session.Types.newSessionId' but is not validated on parse —
   63 -- 'SessionId' is treated as an opaque label so that on-disk session
   64 -- directories created by older or newer code remain readable.
   65 newtype SessionId = SessionId { unSessionId :: Text }
   66   deriving stock (Show, Eq, Ord, Generic)
   67   deriving newtype (Aeson.ToJSON)
   68 
   69 instance Aeson.FromJSON SessionId where
   70   parseJSON = Aeson.withText "SessionId" (pure . SessionId)
   71 
   72 -- | Friendly alias for the 'SessionId' constructor. Provided for
   73 -- symmetry with smart constructors elsewhere; performs no validation.
   74 parseSessionId :: Text -> SessionId

   97 -- | Typed allow-list. @AllowAll@ explicitly opts in to allowing everything.
   98 -- @AllowList s@ restricts to the given set.
   99 data AllowList a
  100   = AllowAll
  101   | AllowList (Set a)
  102   deriving stock (Show, Eq, Generic)
  103 
  104 -- | Check whether a value is permitted by the allow-list.
  105 isAllowed :: Ord a => AllowList a -> a -> Bool
  106 isAllowed AllowAll      _ = True
  107 isAllowed (AllowList s) x = Set.member x s
