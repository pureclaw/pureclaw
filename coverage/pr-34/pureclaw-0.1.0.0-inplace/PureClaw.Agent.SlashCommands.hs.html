  113   , _cs_parse       :: Text -> Maybe SlashCommand
  114     -- ^ Try to parse a stripped, original-case input as this command.

  395 -- | Parse a user message as a slash command.
  396 -- Implemented as 'asum' over '_cs_parse' from 'allCommandSpecs', followed
  397 -- by a catch-all for unrecognised @\/vault@ subcommands.
  398 -- Returns 'Nothing' only for input that does not begin with @\/@.
  399 parseSlashCommand :: Text -> Maybe SlashCommand
  400 parseSlashCommand input =
  401   let stripped = T.strip input
  402   in if "/" `T.isPrefixOf` stripped
  403      then asum (map (`_cs_parse` stripped) allCommandSpecs)
  404             <|> channelUnknownFallback stripped
  405             <|> vaultUnknownFallback stripped
  406             <|> transcriptUnknownFallback stripped
  407             <|> harnessUnknownFallback stripped
  408             <|> agentUnknownFallback stripped
  409             <|> sessionUnknownFallback stripped
  410      else Nothing

  549 -- | Parse "/harness start <name> [dir] [--unsafe]".
  550 -- The first word after "start" is the harness name. Remaining words are
  551 -- split into an optional directory (any non-flag token) and the
  552 -- @--unsafe@ flag.
  553 harnessStartP :: Text -> Maybe SlashCommand
  554 harnessStartP t =
  555   let pfx   = "/harness start"
  556       lower = T.toLower t
  557   in if (pfx <> " ") `T.isPrefixOf` lower
  558      then let rest  = T.strip (T.drop (T.length pfx) t)
  559               (name, afterName) = T.break (== ' ') rest
  560           in if T.null name
  561              then Nothing
  562              else let tokens = T.words (T.strip afterName)
  563                       skipPerms = "--unsafe" `elem` map T.toLower tokens
  564                       positional = filter (\tok -> T.toLower tok /= "--unsafe") tokens
  565                       dir = case positional of
  566                               (d : _) -> Just d
  567                               []      -> Nothing
  568                   in Just (CmdHarness (HarnessStart name dir skipPerms))
  569      else Nothing

 1026 -- | Interactive vault setup wizard. Detects auth mechanisms, lets the user
 1027 -- choose, then creates or rekeys the vault.
 1028 executeVaultSetup :: AgentEnv -> Context -> IO Context
 1029 executeVaultSetup env ctx = do
 1030   let ch   = _env_channel env
 1031       send = _ch_send ch . OutgoingMessage
 1032       ph   = _env_pluginHandle env
 1033 
 1034   -- Step 1: Detect available plugins
 1035   plugins <- _ph_detect ph
 1036 
 1037   -- Step 2: Build choice menu
 1038   let options = buildSetupOptions plugins
 1039       menu    = formatSetupMenu options
 1040   send menu
 1041 
 1042   -- Step 3: Read user's choice
 1043   choiceText <- _ch_prompt ch "Choice: "
 1044   case parseChoice (length options) (T.strip choiceText) of
 1045     Nothing -> do
 1046       send "Invalid choice. Setup cancelled."
 1047       pure ctx
 1048     Just idx -> do
 1049       let chosen = snd (options !! idx)
 1050       -- Step 4: Create encryptor based on choice
 1051       encResult <- createEncryptorForChoice ch ph chosen
 1052       case encResult of
 1053         Left err -> do
 1054           send err
 1055           pure ctx
 1056         Right (newEnc, keyLabel, mRecipient, mIdentity) -> do
 1057           -- Step 5: Init or rekey
 1058           vaultOpt <- readIORef (_env_vault env)
 1059           case vaultOpt of
 1060             Nothing -> do
 1061               -- No vault handle at all: create from scratch
 1062               setupResult <- firstTimeSetup env newEnc keyLabel
 1063               case setupResult of
 1064                 Left err -> send err
 1065                 Right () -> do
 1066                   send ("Vault created with " <> keyLabel <> " encryption.")
 1067                   updateConfigAfterSetup mRecipient mIdentity keyLabel
 1068             Just vault -> do
 1069               -- Vault handle exists — but the file may not.
 1070               -- Try init: if it succeeds, this is first-time setup.
 1071               -- If VaultAlreadyExists, we need to rekey.
 1072               initResult <- _vh_init vault
 1073               case initResult of
 1074                 Right () -> do
 1075                   -- First-time init succeeded (file didn't exist)
 1076                   send ("Vault created with " <> keyLabel <> " encryption.")
 1077                   updateConfigAfterSetup mRecipient mIdentity keyLabel
 1078                 Left VaultAlreadyExists -> do
 1079                   -- Vault exists — rekey it
 1080                   let confirmFn msg = do
 1081                         send msg
 1082                         answer <- _ch_prompt ch "Proceed? [y/N]: "
 1083                         pure (T.strip answer == "y" || T.strip answer == "Y")
 1084                   rekeyResult <- _vh_rekey vault newEnc keyLabel confirmFn
 1085                   case rekeyResult of
 1086                     Left (VaultCorrupted "rekey cancelled by user") ->
 1087                       send "Rekey cancelled."
 1088                     Left err ->
 1089                       send ("Rekey failed: " <> T.pack (show err))
 1090                     Right () -> do
 1091                       send ("Vault rekeyed to " <> keyLabel <> ".")
 1092                       updateConfigAfterSetup mRecipient mIdentity keyLabel
 1093                 Left err ->
 1094                   send ("Vault init failed: " <> T.pack (show err))
 1095           pure ctx

 1763 -- | Discover running harnesses by querying tmux window names.
 1764 -- Returns the reconstructed harness map and the next window index to use.
 1765 --
 1766 -- Window names matching @\<canonical\>-\<N\>@ (e.g. @claude-code-0@) are
 1767 -- recognised as harness windows. The handle is reconstructed so
 1768 -- send\/receive\/stop work against the existing tmux window.
 1769 discoverHarnesses
 1770   :: TranscriptHandle
 1771   -> IO (Map.Map Text HarnessHandle, Int)
 1772 discoverHarnesses = discoverHarnessesIn "pureclaw"
 1773 
 1774 -- | Like 'discoverHarnesses' but queries a specific tmux session name.
 1775 -- Useful for testing with an isolated session.
 1776 discoverHarnessesIn
 1777   :: Text             -- ^ tmux session name
 1778   -> TranscriptHandle
 1779   -> IO (Map.Map Text HarnessHandle, Int)
 1780 discoverHarnessesIn session th = do
 1781   windows <- listSessionWindows session
 1782   let discovered =
 1783         [ (name, idx, canonical)
 1784         | (idx, name) <- windows
 1785         , Just (canonical, winIdx) <- [parseHarnessWindowName name]
 1786         , winIdx == idx  -- sanity: name encodes the same index
 1787         ]
 1788   handles <- mapM (\(name, idx, canonical) -> do
 1789     hh <- mkHandle canonical idx
 1790     pure (name, hh)) discovered
 1791   let harnessMap = Map.fromList handles
 1792       nextIdx = if null discovered
 1793         then 0
 1794         else maximum [idx | (_, idx, _) <- discovered] + 1
 1795   pure (harnessMap, nextIdx)
 1796   where
 1797     mkHandle :: Text -> Int -> IO HarnessHandle
 1798     mkHandle canonical idx = case canonical of
 1799       "claude-code" -> mkDiscoveredClaudeCodeHandle th idx
 1800       -- Future harness types go here
 1801       _             -> mkDiscoveredClaudeCodeHandle th idx  -- fallback
 1802 
 1803     -- | Parse a window name like "claude-code-0" into (canonical, index).
 1804     parseHarnessWindowName :: Text -> Maybe (Text, Int)
 1805     parseHarnessWindowName name =
 1806       -- Try each known canonical name as a prefix
 1807       let candidates =
 1808             [ (canonical, T.drop (T.length canonical + 1) name)
 1809             | (canonical, _, _) <- knownHarnesses
 1810             , (canonical <> "-") `T.isPrefixOf` name
 1811             ]
 1812       in case candidates of
 1813         [(canonical, suffix)] ->
 1814           case TR.decimal suffix of
 1815             Right (n, rest) | T.null rest -> Just (canonical, n)
 1816             _ -> Nothing
 1817         _ -> Nothing

 1874 -- | Link to an existing Signal account by scanning a QR code.
 1875 -- signal-cli link outputs the sgnl:// URI, then blocks until the user
 1876 -- scans it. We need to stream the output to show the URI immediately.
 1877 signalLinkFlow :: AgentEnv -> Context -> IO Context
 1878 signalLinkFlow env ctx = do
 1879   let ch   = _env_channel env
 1880       send = _ch_send ch . OutgoingMessage
 1881 
 1882   send "Generating link... (this may take a moment)"
 1883 
 1884   let procConfig = P.setStdout P.createPipe
 1885                  $ P.setStderr P.createPipe
 1886                  $ P.proc "signal-cli" ["link", "-n", "PureClaw"]
 1887   startResult <- try @IOException $ P.startProcess procConfig
 1888   case startResult of
 1889     Left err -> do
 1890       send $ "Failed to start signal-cli: " <> T.pack (show err)
 1891       pure ctx
 1892     Right process -> do
 1893       let stdoutH = P.getStdout process
 1894           stderrH = P.getStderr process
 1895       -- signal-cli outputs the URI to stderr, then blocks waiting for scan.
 1896       -- Read stderr lines until we find the sgnl:// URI.
 1897       linkUri <- readUntilLink stderrH stdoutH
 1898       case linkUri of
 1899         Nothing -> do
 1900           -- Process may have exited with error
 1901           exitCode <- P.waitExitCode process
 1902           send $ "signal-cli link failed (exit " <> T.pack (show exitCode) <> ")"
 1903           pure ctx
 1904         Just uri -> do
 1905           send $ T.intercalate "\n"
 1906             [ "Open Signal on your phone:"
 1907             , "  Settings \x2192 Linked Devices \x2192 Link New Device"
 1908             , ""
 1909             , "Scan this link (or paste into a QR code generator):"
 1910             , ""
 1911             , "  " <> uri
 1912             , ""
 1913             , "Waiting for you to scan... (this will complete automatically)"
 1914             ]
 1915           -- Now wait for signal-cli to finish (user scans the code)
 1916           exitCode <- P.waitExitCode process
 1917           case exitCode of
 1918             ExitSuccess -> do
 1919               send "Linked successfully!"
 1920               detectAndWriteSignalConfig env ctx
 1921             _ -> do
 1922               send "Link failed or was cancelled."
 1923               pure ctx
 1924   where
 1925     -- Read lines from both handles looking for a sgnl:// URI.
 1926     -- signal-cli typically puts it on stderr.
 1927     readUntilLink :: Handle -> Handle -> IO (Maybe Text)
 1928     readUntilLink stderrH stdoutH = go (50 :: Int)  -- max 50 lines to prevent infinite loop
 1929       where
 1930         go 0 = pure Nothing
 1931         go n = do
 1932           lineResult <- try @IOException (hGetLine stderrH)
 1933           case lineResult of
 1934             Left _ -> do
 1935               -- stderr closed, try stdout
 1936               outResult <- try @IOException (hGetLine stdoutH)
 1937               case outResult of
 1938                 Left _    -> pure Nothing
 1939                 Right line ->
 1940                   let t = T.pack line
 1941                   in if "sgnl://" `T.isInfixOf` t
 1942                      then pure (Just (T.strip t))
 1943                      else go (n - 1)
 1944             Right line ->
 1945               let t = T.pack line
 1946               in if "sgnl://" `T.isInfixOf` t
 1947                  then pure (Just (T.strip t))
 1948                  else go (n - 1)
