   31 -- | Validated session prefix. Used as the human-readable leading segment
   32 -- of a 'PureClaw.Core.Types.SessionId'. Same character rules as
   33 -- 'PureClaw.Agent.AgentDef.AgentName' plus a reserved-word denylist.
   34 newtype SessionPrefix = SessionPrefix { unSessionPrefix :: Text }
   35   deriving stock (Show, Eq, Ord)
   36 
   37 -- | Reasons a raw 'Text' cannot be promoted to a 'SessionPrefix'.
   38 data SessionPrefixError
   39   = PrefixEmpty
   40   | PrefixTooLong
   41   | PrefixInvalidChars Text
   42   | PrefixLeadingDot
   43   | PrefixReserved Text
   44   deriving stock (Show, Eq)
   45 
   46 -- | Maximum allowed length for a session prefix.
   47 sessionPrefixMaxLength :: Int
   48 sessionPrefixMaxLength = 64
   49 
   50 -- | Valid character predicate: ASCII letters, digits, underscore, hyphen.
   51 -- Mirrors 'PureClaw.Agent.AgentDef.isValidAgentNameChar'.
   52 isValidPrefixChar :: Char -> Bool
   53 isValidPrefixChar c =
   54   Char.isAsciiUpper c
   55     || Char.isAsciiLower c
   56     || Char.isDigit c
   57     || c == '_'
   58     || c == '-'
   59 
   60 -- | Reserved tokens that look like prefixes but collide with CLI verbs.
   61 -- Currently just @"new"@, which is the literal argument to
   62 -- @\/session new@ and would create ambiguous resume targets if allowed
   63 -- as a prefix.
   64 reservedPrefixes :: [Text]
   65 reservedPrefixes = ["new"]
   66 
   67 -- | Smart constructor. Same validation as 'mkAgentName' (non-empty,
   68 -- max 64 chars, no leading dot, only @[a-zA-Z0-9_-]@), plus rejection
   69 -- of reserved tokens like @"new"@.
   70 mkSessionPrefix :: Text -> Either SessionPrefixError SessionPrefix
   71 mkSessionPrefix raw
   72   | T.null raw = Left PrefixEmpty
   73   | T.length raw > sessionPrefixMaxLength = Left PrefixTooLong
   74   | T.head raw == '.' = Left PrefixLeadingDot
   75   | not (T.all isValidPrefixChar raw) = Left (PrefixInvalidChars raw)
   76   | raw `elem` reservedPrefixes = Left (PrefixReserved raw)
   77   | otherwise = Right (SessionPrefix raw)
   78 
   79 -- | Custom 'Aeson.FromJSON' routes through 'mkSessionPrefix' so corrupted
   80 -- on-disk JSON cannot bypass the smart constructor.
   81 instance Aeson.FromJSON SessionPrefix where
   82   parseJSON = Aeson.withText "SessionPrefix" $ \t ->
   83     case mkSessionPrefix t of
   84       Right p -> pure p
   85       Left e -> fail ("invalid SessionPrefix: " ++ show e)

   87 -- | Pure session ID generator. Encodes a 'UTCTime' as
   88 -- @YYYYMMDD-HHMMSS-mmm@ (milliseconds zero-padded to three digits)
   89 -- and prefixes it with the optional 'SessionPrefix' separated by a
   90 -- hyphen.
   91 newSessionId :: Maybe SessionPrefix -> UTCTime -> SessionId
   92 newSessionId mPrefix time =
   93   let hms     = formatTime defaultTimeLocale "%Y%m%d-%H%M%S" time
   94       -- Extract milliseconds from the fractional second.
   95       picoDay = diffTimeToPicoseconds (utctDayTime time)
   96       millis  = (picoDay `div` 1000000000) `mod` 1000
   97       timeStr = T.pack hms <> "-" <> T.justifyRight 3 '0' (T.pack (show millis))
   98       full    = case mPrefix of
   99         Nothing -> timeStr
  100         Just p  -> unSessionPrefix p <> "-" <> timeStr
  101   in SessionId full

  103 -- | Whether a session targets the LLM provider directly or a named harness
  104 -- (e.g. an interactive @claude-code@ tmux session).
  105 data RuntimeType
  106   = RTProvider
  107   | RTHarness Text
  108   deriving stock (Show, Eq)
  109 
  110 -- | JSON encoding: @"provider"@ or @"harness:<name>"@. Custom rather than
  111 -- generic so on-disk @session.json@ files stay human-readable and so we
  112 -- don't tie ourselves to aeson's tagged-sum format.
  113 instance Aeson.ToJSON RuntimeType where
  114   toJSON RTProvider       = Aeson.String "provider"
  115   toJSON (RTHarness name) = Aeson.String ("harness:" <> name)
  116 
  117 instance Aeson.FromJSON RuntimeType where
  118   parseJSON = Aeson.withText "RuntimeType" $ \t ->
  119     case t of
  120       "provider" -> pure RTProvider
  121       _ | Just name <- T.stripPrefix "harness:" t -> pure (RTHarness name)
  122         | otherwise -> fail ("Unknown RuntimeType: " <> T.unpack t)
  123 
  124 -- | Map a 'RuntimeType' to its corresponding 'MessageTarget'. Pure helper
  125 -- so the session loader and CLI can share the same default-routing logic
  126 -- without duplicating the case match.
  127 defaultTarget :: RuntimeType -> MessageTarget
  128 defaultTarget RTProvider       = TargetProvider
  129 defaultTarget (RTHarness name) = TargetHarness name

  131 -- | Persistent metadata for a single session. Stored as @session.json@
  132 -- inside the session's on-disk directory.
  133 data SessionMeta = SessionMeta
  134   { _sm_id                :: SessionId
  135   , _sm_agent             :: Maybe AgentName
  136   , _sm_runtime           :: RuntimeType
  137   , _sm_model             :: Text
  138   , _sm_channel           :: Text
  139   , _sm_createdAt         :: UTCTime
  140   , _sm_lastActive        :: UTCTime
  141   , _sm_bootstrapConsumed :: Bool
  142   } deriving stock (Show, Eq, Generic)
  143 
  144 -- Hand-written JSON so we don't depend on a 'ToJSON' instance for
  145 -- 'AgentName' (which 'PureClaw.Agent.AgentDef' deliberately does not
  146 -- export). The agent is serialized as its bare 'unAgentName' string and
  147 -- deserialized via the existing 'FromJSON AgentName' instance, which
  148 -- routes through 'mkAgentName'.
  149 instance Aeson.ToJSON SessionMeta where
  150   toJSON s = Aeson.object $
  151     [ "id"                 .= _sm_id s
  152     , "runtime"            .= _sm_runtime s
  153     , "model"              .= _sm_model s
  154     , "channel"            .= _sm_channel s
  155     , "created_at"         .= _sm_createdAt s
  156     , "last_active"        .= _sm_lastActive s
  157     , "bootstrap_consumed" .= _sm_bootstrapConsumed s
  158     ] <> case _sm_agent s of
  159       Just n  -> ["agent" .= unAgentName n]
  160       Nothing -> []
  161 
  162 instance Aeson.FromJSON SessionMeta where
  163   parseJSON = Aeson.withObject "SessionMeta" $ \o -> SessionMeta
  164     <$> o .:  "id"
  165     <*> o .:? "agent"
  166     <*> o .:  "runtime"
  167     <*> o .:  "model"
  168     <*> o .:  "channel"
  169     <*> o .:  "created_at"
  170     <*> o .:  "last_active"
  171     <*> o .:  "bootstrap_consumed"
